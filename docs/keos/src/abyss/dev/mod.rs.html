14// Even though, there could be more than 4 virtio dev, just set maxium device
15// number to 4. Slot 0: Kernel image. For debugging purpose.
16// Slot 1: Filesystem disk 1.
17static mut BLOCK_DEVS: [Option<Box<dyn BlockOps>>; 4] = [None, None, None, None];
18
19/// Get block device.
20///
21/// - Slot 0: Kernel image. For debugging purpose.
22/// - Slot 1: Filesystem disk 1.
23pub fn get_bdev(slot_idx: usize) -> Option<&'static dyn BlockOps> {
24    unsafe { BLOCK_DEVS.get(slot_idx).and_then(|n| n.as_deref()) }
25}

27/// Sector, an access granuality for the disk.
28#[repr(transparent)]
29#[derive(Debug, Clone, Copy, Eq, PartialEq, Ord, PartialOrd)]
30pub struct Sector(pub usize);
31
32impl Sector {
33    /// Get offset that represented by the sector.
34    #[inline]
35    pub fn into_offset(self) -> usize {
36        self.0 * 512
37    }
38
39    /// Cast into usize.
40    #[inline]
41    pub fn into_usize(self) -> usize {
42        self.0
43    }
44}
45
46impl core::ops::Add<usize> for Sector {
47    type Output = Self;
48
49    fn add(self, rhs: usize) -> Self {
50        Self(self.0 + rhs)
51    }
52}

54pub trait BlockOps {
55    /// Initialize the block device.
56    fn init(&self) -> bool;
57    /// Get total block count of this device.
58    fn block_cnt(&self) -> usize;
59    /// get block size of this device.
60    fn block_size(&self) -> usize;
61    /// Read 512 bytes from disk starting from sector.
62    fn read(&self, sector: Sector, buf: &mut [u8; 512]) -> bool;
63    /// Write 512 bytes to disk starting from sector.
64    fn write(&self, sector: Sector, buf: &[u8; 512]) -> bool;
65    #[doc(hidden)]
66    fn read_block_many(&self, _offset: usize, _buf: &mut [u8]) -> bool {
67        unimplemented!()
68    }
69}
