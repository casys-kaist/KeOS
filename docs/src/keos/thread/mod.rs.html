67impl crate::teletype::Teletype for TtyState {
68    fn write(&mut self, data: &[u8]) -> Result<usize, KernelError> {
69        if let Ok(s) = String::from_utf8(data.to_vec()) {
70            self.output.push_str(&s);
71            Ok(data.len())
72        } else {
73            Err(KernelError::InvalidArgument)
74        }
75    }
76
77    fn read(&mut self, data: &mut [u8]) -> Result<usize, KernelError> {
78        let read_bytes = self.input.len().wrapping_sub(self.idx).min(data.len());
79        data[..read_bytes].copy_from_slice(&self.input[self.idx..self.idx + read_bytes]);
80        self.idx += read_bytes;
81        Ok(read_bytes)
82    }
83}

90static THREAD_STATE_TABLE: SpinLock<BTreeMap<u64, Arc<SpinLock<ThreadState>>>> =
91    SpinLock::new(BTreeMap::new());

101#[unsafe(no_mangle)]
102#[doc(hidden)]
103pub unsafe fn __do_exit(exit_code: i32) -> ! {
104    let _ = abyss::interrupt::InterruptGuard::new();
105    with_current(|th| {
106        let mut et = EXIT_CODE_TABLE.lock();
107        et.remove(&th.tid);
108        et.unlock();
109
110        let mut tst = THREAD_STATE_TABLE.lock();
111        tst.remove(&th.tid);
112        tst.unlock();
113
114        th.exit_status
115            .store(0x8000_0000_0000_0000 | (exit_code as u64), Ordering::SeqCst);
116        let mut state = th.state.lock();
117        *state = ThreadState::Exited(exit_code);
118        state.unlock();
119        scheduler::scheduler().reschedule();
120    });
121    unreachable!()
122}

124/// Check signal for current process and perform an exit when signaled.
125pub(crate) fn __check_for_signal() {
126    let _ = __with_current(|th| {
127        let exit_status = th.exit_status.load(Ordering::SeqCst);
128        if (exit_status & 0x4000_0000_0000_0000) == 0x4000_0000_0000_0000 {
129            unsafe {
130                __do_exit(exit_status as i32);
131            }
132        }
133    });
134}

136/// Kill the thread by specified TID (Thread ID).
137pub fn kill_by_tid(tid: u64, exit_code: i32) -> Result<(), KernelError> {
138    let et = EXIT_CODE_TABLE.lock();
139    let Some(exit_status) = et.get(&tid) else {
140        et.unlock();
141        return Err(KernelError::InvalidArgument);
142    };
143    let exit_status = exit_status.clone();
144    et.unlock();
145
146    debug_assert_eq!(
147        exit_status.load(Ordering::SeqCst) & 0x4000_0000_FFFF_FFFF,
148        0
149    );
150
151    exit_status.store(0x4000_0000_0000_0000 | exit_code as u64, Ordering::SeqCst);
152
153    unsafe {
154        abyss::dev::x86_64::apic::send_ipi(IPIDest::AllExcludingSelf, Mode::Fixed(0x7f));
155    }
156
157    Ok(())
158}

160/// Get specified thread's [`ThreadState`] by TID (Thread ID).
161pub fn get_state_by_tid(tid: u64) -> Result<ThreadState, KernelError> {
162    let tst = THREAD_STATE_TABLE.lock();
163
164    let Some(state) = tst.get(&tid) else {
165        tst.unlock();
166        return Err(KernelError::InvalidArgument);
167    };
168
169    let ts_lock = state.lock();
170    let result = *ts_lock;
171
172    ts_lock.unlock();
173    tst.unlock();
174
175    Ok(result)
176}

179/// An thread abstraction.
180pub struct Thread {
181    /// A stack pointer on context switch.
182    ///
183    /// ## WARNING
184    /// DO NOT CHANGE THE OFFSET THIS FIELDS.
185    /// This offset used in context switch with hard-coded value.
186    /// You must add your own members **BELOWS** this sp field.
187    pub(crate) sp: usize,
188    /// Thread Stack
189    pub(crate) stack: Box<ThreadStack>,
190    /// Thread id
191    pub tid: u64,
192    /// Thread name
193    pub name: String,
194    /// State of the thread.
195    pub state: Arc<SpinLock<ThreadState>>,
196    pub(crate) running_cpu: Arc<AtomicI32>,
197    /// Mixture of exit state (63th and 62th bit) and exit code (lower 32 bits).
198    pub exit_status: Arc<AtomicU64>,
199    #[doc(hidden)]
200    pub task: Option<Box<dyn Task>>,
201    // Grading utils.
202    pub(crate) tty_hook: SpinLock<Option<Arc<SpinLock<TtyState>>>>,
203    pub(crate) allocations: SpinLock<Option<BTreeMap<Kva, &'static Location<'static>>>>,
204}
205
206impl Thread {
207    #[doc(hidden)]
208    pub fn new<I>(name: I) -> Box<Self>
209    where
210        alloc::string::String: core::convert::From<I>,
211    {
212        static TID: AtomicU64 = AtomicU64::new(0);
213        let tid = TID.fetch_add(1, Ordering::SeqCst);
214        let mut stack: Box<ThreadStack> = unsafe { Box::new_uninit().assume_init() };
215        stack.magic = THREAD_MAGIC;
216
217        let exit_status = Arc::new(AtomicU64::new(0));
218        let mut et = EXIT_CODE_TABLE.lock();
219        et.insert(tid, exit_status.clone());
220        et.unlock();
221
222        let state = Arc::new(SpinLock::new(ThreadState::Runnable));
223        let mut tst = THREAD_STATE_TABLE.lock();
224        tst.insert(tid, state.clone());
225        tst.unlock();
226
227        Box::new(Self {
228            sp: 0,
229            stack,
230            tid,
231            name: String::from(name),
232            state,
233            exit_status,
234            running_cpu: Arc::new(AtomicI32::new(-1)),
235            task: None,
236            tty_hook: SpinLock::new(
237                __with_current(|th| {
238                    let guard = th.tty_hook.lock();
239                    let val = guard.as_ref().map(|n| n.clone());
240                    guard.unlock();
241                    val
242                })
243                .unwrap_or(None),
244            ),
245            allocations: SpinLock::new(None),
246        })
247    }
248
249    #[doc(hidden)]
250    pub fn track_alloc(&self) {
251        let mut guard = self.allocations.lock();
252        *guard = Some(BTreeMap::new());
253        guard.unlock();
254    }
255
256    #[doc(hidden)]
257    pub fn validate_alloc(&self) {
258        let guard = self.allocations.lock();
259        if let Some(allocs) = guard.as_ref()
260            && !allocs.is_empty()
261        {
262            struct DebugAlloc<'a>(&'a BTreeMap<Kva, &'static Location<'static>>);
263            impl core::fmt::Debug for DebugAlloc<'_> {
264                fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
265                    writeln!(f, "List of unallocated pages:")?;
266                    for (idx, (va, loc)) in self.0.iter().take(10).enumerate() {
267                        writeln!(f, "  {idx}: {va:?}, allocated at {loc}")?;
268                    }
269                    if self.0.len() > 10 {
270                        write!(f, "... and {:?} more allocations.", self.0.len() - 10)?;
271                    }
272                    Ok(())
273                }
274            }
275            panic!(
276                "Grader: Validating `Page` allocation state failed: Detecting {} non-freed pages.\n{:?}",
277                allocs.len(),
278                DebugAlloc(allocs)
279            );
280        }
281        guard.unlock();
282    }
283
284    pub(crate) unsafe fn do_run(&mut self) {
285        unsafe {
286            let _p = abyss::interrupt::InterruptGuard::new();
287            let next_sp = self.sp;
288            let current_sp = with_current(|th| {
289                while self.running_cpu.load(Ordering::SeqCst) != -1 {
290                    core::hint::spin_loop()
291                }
292                &mut th.sp as *mut usize
293            });
294            assert_eq!(
295                abyss::interrupt::InterruptState::current(),
296                abyss::interrupt::InterruptState::Off
297            );
298            context_switch_trampoline(current_sp, next_sp)
299        }
300    }
301
302    pub(crate) fn run(self: Box<Self>) {
303        unsafe { Box::into_raw(self).as_mut().unwrap().do_run() }
304    }

306    /// Pin current thread not to be scheduled by blocking interrupt.
307    ///
308    /// When [`ThreadPinGuard`] is dropped, the current thread is unpinned.
309    /// When you hold multiple [`ThreadPinGuard`], you **MUST** drops
310    /// [`ThreadPinGuard`] as a reverse order of creation.
311    pub fn pin() -> ThreadPinGuard {
312        ThreadPinGuard::new()
313    }
314
315    #[doc(hidden)]
316    pub fn hook_stdin(&self, b: &'static [u8]) {
317        let mut guard = self.tty_hook.lock();
318        if guard.is_some() {
319            panic!("Fail to hook stdin: already hook.");
320        } else {
321            *guard = Some(Arc::new(SpinLock::new(TtyState {
322                input: b,
323                idx: 0,
324                output: String::new(),
325            })));
326            guard.unlock();
327        }
328    }
329
330    #[doc(hidden)]
331    pub fn finish_hook(&self) -> Option<String> {
332        let mut guard = self.tty_hook.lock();
333        let val = guard.take().map(|n| {
334            let guard = n.lock();
335            let val = alloc::borrow::ToOwned::to_owned(&(*guard.output));
336            guard.unlock();
337            val
338        });
339        guard.unlock();
340        val
341    }

347/// A handle to join thread.
348pub struct JoinHandle
349where
350    Self: 'static,
351{
352    /// Thread id of this handle.
353    pub tid: u64,
354    exit_status: Arc<AtomicU64>,
355    running_cpu: Arc<AtomicI32>,
356}
357
358impl JoinHandle {
359    /// Make a join handle for Thread `th`.
360    pub fn new_for(th: &Thread) -> Self {
361        Self {
362            tid: th.tid,
363            exit_status: th.exit_status.clone(),
364            running_cpu: th.running_cpu.clone(),
365        }
366    }
367
368    /// Join this handle and returns exit code.
369    pub fn join(self) -> i32 {
370        loop {
371            let v = self.exit_status.load(Ordering::SeqCst);
372            if v >= 0x8000_0000_0000_0000 {
373                return v as i32;
374            }
375            crate::scheduler().reschedule();
376        }
377    }
378
379    /// Get scheudled cpu id of the underlying thread.
380    ///
381    /// If the thread is not runnig, returns None.
382    pub fn try_get_running_cpu(&self) -> Option<usize> {
383        match self.running_cpu.load(Ordering::SeqCst) {
384            v if v < 0 => None,
385            v => Some(v as usize),
386        }
387    }
388}

393/// A handle that represent the parked thread.
394pub struct ParkHandle {
395    pub(crate) th: Box<Thread>,
396}
397
398impl ParkHandle {
399    pub(crate) fn new_for(th: Box<Thread>) -> Self {
400        Self { th }
401    }
402
403    /// Consume the handle and unpark the underlying thread.
404    pub fn unpark(self) {
405        // Wait until context switch is finished.
406        while self.th.running_cpu.load(Ordering::SeqCst) != -1 {
407            core::hint::spin_loop()
408        }
409        let mut state = self.th.state.lock();
410        *state = ThreadState::Runnable;
411        state.unlock();
412
413        scheduler::scheduler().push_to_queue(self.th);
414    }
415}

422/// The context-switch magic.
423#[unsafe(naked)]
424unsafe extern "C" fn context_switch_trampoline(_current_sp: *mut usize, _next_sp: usize) {
425    // XXX: we don't need to rflags because when threads entered this function the
426    // rflags state is always same. RDI: Current Stack pointer storage.
427    // RSI: Next Stack pointer.
428    naked_asm!(
429        "push rbp",
430        "push rbx",
431        "push r12",
432        "push r13",
433        "push r14",
434        "push r15",
435
436        // Switch.
437        "mov r8, rsp",
438        "mov [rdi], r8",
439        "mov rsp, rsi",
440
441        "pop r15",
442        "pop r14",
443        "pop r13",
444        "pop r12",
445        "pop rbx",
446        "pop rbp",
447
448        // XXX: Tail-call optimization, pass prev thread to rdi
449        "jmp {}",
450        sym finish_context_switch
451    );
452}

454unsafe extern "C" fn finish_context_switch(prev: &'static mut Thread) {
455    unsafe {
456        assert_eq!(
457            abyss::interrupt::InterruptState::current(),
458            abyss::interrupt::InterruptState::Off
459        );
460
461        let prev_state = {
462            let lock = prev.state.lock();
463            let result = *lock;
464            lock.unlock();
465            result
466        };
467
468        let _dropped = match prev_state {
469            ThreadState::Exited(_e) => Some(Box::from_raw(prev)),
470            ThreadState::Idle => None,
471            ThreadState::Running => {
472                let mut prev_state = prev.state.lock();
473                *prev_state = ThreadState::Runnable;
474                prev_state.unlock();
475
476                let th = Box::from_raw(prev);
477                scheduler::scheduler().push_to_queue(th);
478                None
479            }
480            ThreadState::Parked => None,
481            ThreadState::Runnable => unreachable!("{:?} {:?}", prev as *const _, prev.name),
482        };
483        with_current(|th| {
484            let mut state = th.state.lock();
485            if *state != ThreadState::Idle {
486                *state = ThreadState::Running
487            }
488            state.unlock();
489
490            __check_for_signal();
491
492            abyss::x86_64::segmentation::SegmentTable::update_tss(
493                th.stack.as_mut() as *mut _ as usize + STACK_SIZE,
494            );
495            th.running_cpu.store(cpuid() as i32, Ordering::SeqCst);
496
497            if let Some(task) = th.task.as_mut() {
498                task.with_page_table_pa(&(load_pt as fn(Pa)));
499            } else {
500                unsafe extern "C" {
501                    static mut boot_pml4e: u64;
502                }
503                load_pt(Pa::new(boot_pml4e as usize).unwrap());
504            }
505        });
506        prev.running_cpu.store(-1, Ordering::SeqCst);
507        drop(_dropped);
508    }
509}

511#[inline]
512pub(crate) fn __with_current<R>(f: impl FnOnce(&mut Thread) -> R) -> Result<R, usize> {
513    unsafe {
514        let mut sp: usize;
515        asm!("mov {}, rsp", out(reg) sp);
516        if let Some(stack) = ((sp & !(STACK_SIZE - 1)) as *mut ThreadStack).as_mut()
517            && stack.magic == THREAD_MAGIC
518        {
519            return Ok(f(stack.thread.as_mut().unwrap()));
520        }
521        Err(sp)
522    }
523}

530impl Current {
531    /// Run a function `f` with [`ParkHandle`] for current thread, and then park
532    /// the current thread.
533    pub fn park_with(f: impl FnOnce(ParkHandle)) {
534        with_current(|th| {
535            f(unsafe { scheduler::scheduler().park_thread(th).unwrap() });
536        });
537        assert!(
538            abyss::interrupt::InterruptState::current() == abyss::interrupt::InterruptState::On,
539            "Try to park a thread while holding a lock."
540        );
541        let _ = abyss::interrupt::InterruptGuard::new();
542        scheduler::scheduler().reschedule();
543    }

545    /// Exit the current thread with `exit_code`.
546    pub fn exit(exit_code: i32) -> ! {
547        assert!(
548            abyss::interrupt::InterruptState::current() == abyss::interrupt::InterruptState::On,
549            "Try to exit a thread while holding a lock."
550        );
551        unsafe {
552            __do_exit(exit_code);
553        }
554    }
555
556    /// Get the current thread's id.
557    pub fn get_tid() -> u64 {
558        with_current(|th| th.tid)
559    }

627
628    fn into_thread<F: FnOnce() + Send + 'static>(self, thread_fn: F) -> Box<Thread> {
629        /// The very beginning of the thread
630        #[unsafe(naked)]
631        unsafe extern "C" fn start<F: FnOnce() + Send>() -> ! {
632            naked_asm!(
633                "pop rdi",
634                "sti",
635                "jmp {}",
636                sym thread_start::<F>
637            );
638        }
639
640        fn thread_start<F: FnOnce() + Send>(thread_fn: *mut F) {
641            unsafe {
642                core::intrinsics::catch_unwind(
643                    |o| {
644                        let o = *Box::from_raw(o as *mut F);
645                        o();
646                        Current::exit(0);
647                    },
648                    thread_fn as *mut u8,
649                    |_, _| {},
650                );
651            }
652        }
653
654        let Self { mut th } = self;
655        let stack = th.stack.as_mut();
656        let frame = unsafe {
657            ((&mut stack._usable_marker as *mut _ as usize
658                - core::mem::size_of::<ContextSwitchFrame<F>>())
659                as *mut ContextSwitchFrame<F>)
660                .as_mut()
661                .unwrap()
662        };
663        frame.end_of_stack = 0;
664        frame.thread_fn = Box::into_raw(Box::new(thread_fn));
665        frame.ret_addr = start::<F> as usize;
666        th.sp = frame as *mut _ as usize;
667        th.stack.thread = th.as_mut() as *mut _;
668        th
669    }
