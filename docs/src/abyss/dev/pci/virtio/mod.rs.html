19pub trait VirtIoDeviceFeature {
20    fn bits(&self) -> u64;
21    fn from_bits_truncate(val: u64) -> Self;
22}

24pub struct VirtIoDevice<V: Send + Sync, const MAX_QUEUE: usize> {
25    virtqs: [SpinLock<VirtQueue>; MAX_QUEUE],
26    pub transport: PciTransport<V>,
27}
28
29impl<V: Send + Sync, const MAX_QUEUE: usize> VirtIoDevice<V, MAX_QUEUE> {
30    #[inline]
31    pub fn from_transport(transport: PciTransport<V>) -> Self {
32        Self {
33            transport,
34            virtqs: [0; MAX_QUEUE].map(|_| SpinLock::new(VirtQueue::empty())),
35        }
36    }
37
38    #[inline]
39    pub fn init<F, FN>(
40        &self,
41        common_feature_mask: VirtIoFeaturesCommon,
42        device_feature_mask: F,
43        device_init: FN,
44    ) -> Result<(), VirtIOError>
45    where
46        F: VirtIoDeviceFeature,
47        FN: FnOnce(&Self, VirtIoFeaturesCommon, F) -> Result<(), VirtIOError>,
48    {
49        // 3.1.1 Driver Requirements: Device Initialization
50        let status = self.transport.common.device_status();
51        // reset device.
52        status.write(Status::empty());
53
54        // Wait until read returns zero.
55        loop {
56            match status.read() {
57                status if status.is_empty() => break,
58                status if status.contains(Status::FAILED) => return Err(VirtIOError),
59                _ => (),
60            }
61        }
62        // Step 2-3.
63        status.write(Status::ACKNOWLEDGE | Status::DRIVER);
64        // Step 4. Ack features
65        let features = self.transport.common.get_device_features();
66        let common_features = VirtIoFeaturesCommon::from_bits_truncate(features)
67            & common_feature_mask
68            | VirtIoFeaturesCommon::VERSION_1;
69        let device_features = F::from_bits_truncate(features & device_feature_mask.bits());
70        self.transport.feat.store(
71            common_features.bits() | device_features.bits(),
72            Ordering::Relaxed,
73        );
74        self.transport
75            .common
76            .set_features(common_features.bits() | device_features.bits());
77        // Step 5.
78        status.write(status.read() | Status::FEATURES_OK);
79        // Step 6.
80        if !status.read().contains(Status::FEATURES_OK) {
81            return Err(VirtIOError);
82        }
83        // Step 7. Device-specific setups
84        device_init(self, common_features, device_features)?;
85        // Step 8.
86        status.write(status.read() | Status::DRIVER_OK);
87        Ok(())
88    }
89
90    #[inline]
91    pub fn configure_queue<F, R>(&self, qid: u16, f: F) -> R
92    where
93        F: FnOnce(QueueScope<V, MAX_QUEUE>) -> R,
94    {
95        assert!(qid < MAX_QUEUE as u16);
96        self.transport.select_queue(qid);
97        f(QueueScope { qid, dev: self })
98    }
99
100    #[inline]
101    pub fn get_queue(&self, qid: u16) -> Option<SpinLockGuard<'_, VirtQueue>> {
102        self.virtqs.get(qid as usize).map(|n| n.lock())
103    }
104}

106pub struct QueueScope<'a, V: Send + Sync, const MAX_QUEUE: usize> {
107    qid: u16,
108    dev: &'a VirtIoDevice<V, MAX_QUEUE>,
109}

111impl<'dev, 'qs, V: Send + Sync, const MAX_QUEUE: usize> QueueScope<'dev, V, MAX_QUEUE> {
112    #[inline]
113    pub fn queue_builder(&'qs self) -> QueueBuilder<'dev, 'qs, V, MAX_QUEUE> {
114        QueueBuilder {
115            scope: self,
116            size: 0,
117        }
118    }
119
120    #[inline]
121    pub fn queue_size(&self) -> u16 {

123    }
124}
125
126pub struct QueueBuilder<'a, 'b, V: Send + Sync, const MAX_QUEUE: usize> {
127    scope: &'b QueueScope<'a, V, MAX_QUEUE>,
128    size: u16,
129}
130
131impl<V: Send + Sync, const MAX_QUEUE: usize> QueueBuilder<'_, '_, V, MAX_QUEUE> {
132    #[inline]
133    pub fn set_size(mut self, size: u16) -> Result<Self, VirtIOError> {
134        if size.is_power_of_two() {
135            self.size = size;
136            Ok(self)
137        } else {
138            Err(VirtIOError)
139        }
140    }
141
142    #[inline]
143    pub fn register(self) {
144        let mut guard = self.scope.dev.virtqs[self.scope.qid as usize].lock();
145        let kick = self.scope.dev.transport.get_kick();
146        *guard = VirtQueue::new(
147            self.size,
148            self.scope.qid,
149            VirtIoFeaturesCommon::from_bits_truncate(
150                self.scope.dev.transport.get_driver_features(),
151            )
152            .contains(VirtIoFeaturesCommon::RING_EVENT_IDX),
153            kick,
154        );
155        unsafe {
156            self.scope.dev.transport.register_virtqueue(&guard);
157        }
158        guard.unlock();
159    }
