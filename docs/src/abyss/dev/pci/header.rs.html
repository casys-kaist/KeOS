6bitflags::bitflags! {
7    /// Pci device status.
8    pub struct Status: u16 {
9        /// This bit will be set to 1 whenever the device detects a parity error, even if parity error handling is disabled.
10        const DETECTED_PARITY_ERROR = 1 << 15;
11        /// This bit will be set to 1 whenever the device asserts SERR#.
12        const SIGNALLED_SYSTEM_ERROR = 1 << 14;
13        /// This bit will be set to 1, by a master device, whenever its transaction (except for Special Cycle transactions) is terminated with Master-Abort.
14        const RECEIVED_MASTER_ABORT = 1 << 13;
15        /// This bit will be set to 1, by a master device, whenever its transaction is terminated with Target-Abort.
16        const RECEIVED_TARGET_ABORT = 1 << 12;
17        /// This bit will be set to 1 whenever a target device terminates a transaction with Target-Abort.
18        const SIGNALLED_TARGET_ABORT = 1 << 11;
19        /// Read only bits that represent the slowest time that a device will assert DEVSEL# for any bus command except Configuration Space read and writes. Where a value of 0x00 represents fast timing, a value of 0x01 represents medium timing, and a value of 0x02 represents slow timing.
20        const DEVSEL_TIMING = 1 << 10;
21        /// This bit is only set when the following conditions are met. The bus agent asserted PERR# on a read or observed an assertion of PERR# on a write, the agent setting the bit acted as the bus master for the operation in which the error occurred, and bit 6 of the Command register (Parity Error Response bit) is set to 1.
22        const MASTER_DATA_PARITY_ERROR = 1 << 9;
23        /// If set to 1 the device can accept fast back-to-back transactions that are not from the same agent; otherwise, transactions can only be accepted from the same agent.
24        const FAST_BACK_TO_BACK_CAPABLE = 1 << 8;
25        /// If set to 1 the device is capable of running at 66 MHz; otherwise, the device runs at 33 MHz.
26        const MHZ66_CAPABLE = 1 << 5;
27        /// If set to 1 the device implements the pointer for a New Capabilities Linked list at offset 0x34; otherwise, the linked list is not available.
28        const CAPABILITIES_LIST = 1 << 4;
29        /// Represents the state of the device's INTx# signal. If set to 1 and bit 10 of the Command register (Interrupt Disable bit) is set to 0 the signal will be asserted; otherwise, the signal will be ignored
30        const INTERRUPT_STATUS = 1 << 3;
31    }

51/// Generic implementation of PciHeaderType.
52#[derive(Debug, Clone, Copy)]
53pub struct PciHeader<const V: usize> {
54    pub(crate) pci_device: PciDevice,
55    pub(crate) function: u8,
56}
57
58impl<const V: usize> PciHeader<V> {
59    #[inline(always)]
60    pub fn accessor(&self, off: u8) -> PciAccessor {
61        PciAccessor::new(
62            self.pci_device.bus,
63            self.pci_device.device,
64            self.function,
65            off,
66        )
67    }
68}

70impl PciHeader<0> {
71    /// Get status of the device.
72    #[inline]
73    pub fn status(&self) -> Status {
74        Status::from_bits_truncate((self.accessor(0x4).read_u32() >> 16) as u16)
75    }
76
77    /// Get iterator for enumerating the capabilties of device.
78    #[inline]
79    pub fn capabilities(&self) -> CapabilityIterator<'_, 0> {
80        CapabilityIterator {
81            next: if self.status().contains(Status::CAPABILITIES_LIST) {
82                self.accessor(0x34).read_u8()
83            } else {
84                0
85            },
86            pci_header: self,
87        }
88    }
89
90    /// Get BAR of the device.
91    #[inline]
92    pub fn bar(&self, index: u8) -> Option<Bar> {
93        if index > 5 {
94            return None;
95        }
96        let bar_accessor = self.accessor(0x10 + 4 * index);
97        let bar = bar_accessor.read_u32();
98
99        // Resolve length.
100        bar_accessor.write_u32(u32::MAX);
101        let bar_length = bar_accessor.read_u32();
102        bar_accessor.write_u32(bar);
103
104        if bar & 1 == 1 {
105            Some(Bar::IoSpace(IoSpace {
106                base: bar & !3,
107                length: !(bar_length & !0b11) + 1,
108            }))
109        } else {
110            let (prefetchable, ty) = (bar & 8 == 8, (bar >> 1) & 3);
111            let addr = match ty {
112                0 => (bar & !0xf) as usize,
113                2 if index <= 4 => {
114                    (((bar & !0xf) as u64)
115                        | ((self.accessor(0x10 + 4 * index + 4).read_u32() as u64) << 32))
116                        as usize
117                }
118                // 1 => reserved.
119                _ => return None,
120            };
121            Pa::new(addr).map(|base| {
122                Bar::MemorySpace(MemorySpace {
123                    base,
124                    length: (!(bar_length & !0b1111) + 1) as usize,
125                    _prefetchable: prefetchable,
126                })
127            })
128        }
129    }
