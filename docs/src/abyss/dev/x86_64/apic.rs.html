1//! Advanced Programmable Interrupt Controller (APIC) driver.
2//!
3//! This implements X2Apic mode.
4use crate::dev::DeviceError;
5use crate::x86_64::{msr::Msr, pio::Pio};
6use core::convert::TryFrom;
7
8enum MapDest {
9    Master(u8),
10    Slave(u8),
11}
12
13impl TryFrom<u8> for MapDest {
14    type Error = ();
15
16    fn try_from(value: u8) -> Result<Self, Self::Error> {
17        if value < 32 {
18            Err(())
19        } else if value < 40 {
20            Ok(Self::Master(value - 32))
21        } else if value < 48u8 {
22            Ok(Self::Slave(value - 40))
23        } else {
24            Err(())
25        }
26    }
27}

29/// 8259A interrupt controller
30pub struct _8259A;
31
32impl _8259A {
33    const MASK: u16 = 0b1111_1111_1111_1111;
34
35    pub(crate) fn init() {
36        // mask all of 8259A
37        Pio::new(0x21).write_u8(0xff);
38        Pio::new(0xa1).write_u8(0xff);
39
40        // ICW1: select 8259A-1 init
41        Pio::new(0x20).write_u8(0x11);
42        // ICW2: 8259A-1 IR0-7 mapped to 32.
43        Pio::new(0x21).write_u8(32);
44        // ICW3: 8259A-1 (the master) has a slave on IR2
45        Pio::new(0x21).write_u8(1 << 2);
46        // ICW4: master does Auto EOI
47        Pio::new(0x21).write_u8(3);
48
49        // Setup slave
50        // ICW1: select 8259A-2 init
51        Pio::new(0xa0).write_u8(0x11);
52        // ICW2: 8259A-2 IR0-7 mapped to 40.
53        Pio::new(0xa1).write_u8(40);
54        // ICW3: 8259A-2 is a slave on master's IR2
55        Pio::new(0xa1).write_u8(2);
56        // ICW4: slave's support for AEOI in flat mode is to be investigated.
57        Pio::new(0xa1).write_u8(1);
58
59        // clear specific mask.
60        Pio::new(0x20).write_u8(0x68);
61        // read IRR by default.
62        Pio::new(0x20).write_u8(0x0a);
63
64        Pio::new(0xa0).write_u8(0x68);
65        Pio::new(0xa0).write_u8(0x0a);
66
67        // restore master IRQ mask.
68        Pio::new(0x21).write_u8(Self::MASK as u8);
69        // restore slave IRQ mask.
70        Pio::new(0xa1).write_u8((Self::MASK >> 8) as u8);
71    }
72
73    #[allow(dead_code)]
74    pub(crate) fn enable(ev: u8) -> Result<(), ()> {
75        MapDest::try_from(ev).map(|dest| {
76            let (port, mask) = match dest {
77                MapDest::Master(n) => (Pio::new(0x21), 1 << n),
78                MapDest::Slave(n) => (Pio::new(0xa1), 1 << n),
79            };
80            port.write_u8(port.read_u8() & !mask)
81        })
82    }
83
84    #[allow(dead_code)]
85    pub(crate) fn disable(ev: u8) -> Result<(), ()> {
86        MapDest::try_from(ev).map(|dest| {
87            let (port, mask) = match dest {
88                MapDest::Master(n) => (Pio::new(0x21), 1 << n),
89                MapDest::Slave(n) => (Pio::new(0xa1), 1 << n),
90            };
91            port.write_u8(port.read_u8() | mask)
92        })
93    }
94}

96pub unsafe fn init(core_id: usize) -> Result<(), DeviceError> {
97    unsafe {
98        if core::arch::x86_64::__cpuid(1).ecx & (1 << 21) != 0 {
99            // Enable the x2 apic
100            let apic_base = Msr::<0x1b>::read();
101            Msr::<0x1b>::write(apic_base | (1 << 10));
102            // Enable local apic and set susprious irq vector.
103            // IRQ_SUSPRIOUS = 0xff;
104            // SIV
105            Msr::<0x80f>::write(0x100 | 0xff);
106            // TP
107            Msr::<0x808>::write((Msr::<0x808>::read() & 0xff) | 0x10);
108            // lint1 = MASK | NMI
109            // Msr::<0x836>::write(0x10000 | 0x400);
110            if core_id == 0 {
111                // lint0
112                Msr::<0x835>::write(0x700);
113                _8259A::init();
114            } else {
115                // lint0
116                // MASK | ExtInt
117                Msr::<0x835>::write(0x10000 | 0x700);
118            }
119            Ok(())
120        } else {
121            Err(DeviceError("X2Apic is not supported."))
122        }
123    }
124}

126pub fn eoi() {
127    unsafe {
128        Msr::<0x80b>::write(0);
129    }

132pub enum IPIDest {
133    Cpu(usize),
134    _Self,
135    AllIncludingSelf,
136    AllExcludingSelf,
137}
138
139pub enum Mode {
140    Fixed(u8),
141    Smi,
142    Nmi,
143    Init,
144    StartUp(u8),
145}
146
147pub unsafe fn send_ipi(dest: IPIDest, mode: Mode) {
148    unsafe {
149        Msr::<0x830>::write(
150            match dest {
151                IPIDest::Cpu(cpu) => (cpu as u64) << 32,
152                IPIDest::_Self => 0b01 << 18,
153                IPIDest::AllIncludingSelf => 0b10 << 18,
154                IPIDest::AllExcludingSelf => 0b11 << 18,
155            } | match mode {
156                Mode::Fixed(v) => v as u64,
157                Mode::Smi => 0b010 << 8,
158                Mode::Nmi => 0b100 << 8,
159                Mode::Init => 0b101 << 8,
160                Mode::StartUp(v) => (0b110 << 8) | (v as u64),
161            } | 0x4000,
162        );
163    }
