34pub enum ExceptionHandlingPhase {
35    Search,
36    Cleanup,
37}

39#[allow(dead_code)]
40#[derive(Clone, Copy)]
41enum PersonalityResult {
42    Continue,
43    Run(usize),
44    Error,
45    Stop,
46}

48pub enum UnwindError {
49    BadRegister,
50    BadOpcode(u8),
51    BadOperand(u8),
52    InvalidOp(u8),
53    InvalidApplication,
54    InvalidPc(usize),
55    UnknownRegister,
56    UnmanagedRegister,
57    ParsingFailure,
58    UnwindablePc(usize),
59    MemoryOutOfBound(usize, core::ops::Range<usize>),
60    PersonalityFailure,
61}
62
63impl core::fmt::Debug for UnwindError {
64    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
65        match self {
66            Self::BadRegister => write!(f, "BadRegister"),
67            Self::BadOpcode(v) => write!(f, "BadOpcode(0x{v:x})"),
68            Self::BadOperand(v) => write!(f, "BadOperand(0x{v:x})"),
69            Self::InvalidOp(v) => write!(f, "InvalidOp(0x{v:x})"),
70            Self::InvalidApplication => write!(f, "InvalidApplication"),
71            Self::InvalidPc(v) => write!(f, "InvalidPc(0x{v:x})"),
72            Self::UnknownRegister => write!(f, "UnknownRegister"),
73            Self::UnmanagedRegister => write!(f, "UnmanagedRegister"),
74            Self::ParsingFailure => write!(f, "ParsingFailure"),
75            Self::UnwindablePc(v) => write!(f, "UnwindablePc(0x{v:x})"),
76            Self::MemoryOutOfBound(v, _) => write!(f, "MemoryOutOfBound(0x{v:x})"),
77            Self::PersonalityFailure => write!(f, "PersonalityFailure"),
78        }
79    }
80}

82#[derive(Clone)]
83#[repr(C)]
84pub struct UnwindBacktrace<T>
85where
86    T: Peeker,
87{
88    pub frame: StackFrame,
89    sp_range: core::ops::Range<usize>,
90    cfa: usize,
91    reader: DwarfReader<T>,
92}
93
94impl<T> UnwindBacktrace<T>
95where
96    T: Peeker,
97{
98    #[inline(never)]
99    pub fn new(
100        frame: StackFrame,
101        sp_range: core::ops::Range<usize>,
102        reader: DwarfReader<T>,
103    ) -> Self {
104        let cfa = frame.sp();
105        UnwindBacktrace {
106            frame,
107            sp_range,
108            cfa,
109            reader,
110        }
111    }
112
113    #[inline]
114    pub fn read_mem(&self, addr: usize) -> Result<usize, UnwindError> {
115        if self.sp_range.contains(&addr) {
116            Ok(unsafe { (addr as *const usize).as_ref() }.cloned().unwrap())
117        } else {
118            Err(UnwindError::MemoryOutOfBound(addr, self.sp_range.clone()))
119        }
120    }
121
122    #[inline]
123    fn do_unwind_frame<UnwindFn>(mut self, mut unwind_fn: UnwindFn) -> Result<(), UnwindError>
124    where
125        UnwindFn: FnMut(Self, &FrameDescriptionEntry) -> (Self, bool),
126    {
127        let hdr = EhFrameHeader::parse(self.reader.clone());
128        let (mut previous_pc, mut previous_cfa) = (self.frame.pc(), self.cfa);
129        while self.frame.pc() != 0 {
130            let fde = hdr
131                .find(self.frame.pc())
132                .unwrap()
133                .insn
134                .parse(self.reader.clone())
135                .ok_or(UnwindError::ParsingFailure)?;
136            if fde.pc.contains(&self.frame.pc()) || (self.frame.pc() >> 16 == 0x40)
137            /* "Known" PC for userprog */
138            {
139                let (s, is_stop) = unwind_fn(self, &fde);
140                if is_stop {
141                    return Ok(());
142                }
143                self = s;
144                fde.run(self.frame.pc())?.apply(&mut self)?;
145                if self.frame.pc() == previous_pc && self.cfa == previous_cfa {
146                    return Err(UnwindError::UnwindablePc(self.frame.pc()));
147                }
148
149                self.frame.set_pc(self.frame.pc().wrapping_sub(1));
150                previous_pc = self.frame.pc();
151                previous_cfa = self.cfa;
152            } else {
153                return Err(UnwindError::InvalidPc(self.frame.pc()));
154            }
155        }
156        Ok(())
157    }
158
159    #[inline]
160    pub fn unwind_frame<UnwindFn>(self, mut unwind_fn: UnwindFn) -> Result<(), UnwindError>
161    where
162        UnwindFn: FnMut(&Self, &FrameDescriptionEntry),
163    {
164        self.do_unwind_frame(|this, fde| {
165            unwind_fn(&this, fde);
166            (this, false)
167        })
168    }
169
170    /// Raise exception through unwind with hook function.
171    ///
172    /// # Safety
173    /// LSDA in Unwind table must point the valid handler address.
174    #[inline]
175    pub unsafe fn run<S, BacktraceFn>(
176        self,
177        state: &mut S,
178        mut backtrace: BacktraceFn,
179    ) -> Result<(), UnwindError>
180    where
181        BacktraceFn: FnMut(&mut S, &Self, &FrameDescriptionEntry),
182    {
183        unsafe {
184            let mut action_after_search = PersonalityResult::Continue;
185
186            self.do_unwind_frame(|this, fde| {
187                backtrace(state, &this, fde);
188                // Call personality if available.
189                if let Some(personality) = fde.cie.personality {
190                    let routine = personality
191                        as *const fn(
192                            ExceptionHandlingPhase,
193                            &FrameDescriptionEntry,
194                            &StackFrame,
195                        ) -> PersonalityResult;
196                    match (*routine)(ExceptionHandlingPhase::Search, fde, &this.frame) {
197                        PersonalityResult::Error => action_after_search = PersonalityResult::Error,
198                        PersonalityResult::Stop => return (this, true),
199                        _ => (),
200                    }
201                }
202                (this, false)
203            })
204        }
205    }
