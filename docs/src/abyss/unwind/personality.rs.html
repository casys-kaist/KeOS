17use crate::unwind::{
18    DwarfReader, Encoding, ExceptionHandlingPhase, FrameDescriptionEntry, Peeker,
19    PersonalityResult, StackFrame,
20};
21#[derive(Debug)]
22enum EhAction {
23    None,
24    Cleanup(usize),
25    Catch(usize),
26}
27use core::arch::naked_asm;
28
29#[lang = "eh_personality"]
30//#[unsafe(no_mangle)]
31#[doc(hidden)]
32fn rust_eh_personality(
33    phase: ExceptionHandlingPhase,
34    fde: &FrameDescriptionEntry,
35    frame: &mut StackFrame,
36) -> PersonalityResult {
37    if let Some(action) = find_eh_action(fde, frame) {
38        if matches!(phase, ExceptionHandlingPhase::Search) {
39            match action {
40                EhAction::None | EhAction::Cleanup(_) => PersonalityResult::Continue,
41                EhAction::Catch(_) => PersonalityResult::Stop,
42            }
43        } else {
44            match action {
45                EhAction::None => PersonalityResult::Continue,
46                EhAction::Cleanup(lpad) | EhAction::Catch(lpad) => PersonalityResult::Run(lpad),
47            }
48        }
49    } else {
50        PersonalityResult::Error
51    }
52}
53
54fn find_eh_action(fde: &FrameDescriptionEntry, frame: &StackFrame) -> Option<EhAction> {
55    #[derive(Clone)]
56    struct Reader;
57    impl Peeker for Reader {
58        fn read<T>(&self, ofs: usize) -> Option<T>
59        where
60            T: Copy,
61        {
62            unsafe { (ofs as *const T).as_ref().cloned() }
63        }
64    }
65
66    if let Some(lsda) = fde.lsda {
67        let mut reader = DwarfReader::from_peeker(lsda, Reader);
68        let lpad_base = match reader.read::<u8>()? {
69            0xff => fde.pc.start,
70            encoding => reader.read_with_encoding(Encoding::from(encoding))?,
71        };
72        let ttype_encoding = reader.read::<u8>()?;
73        if ttype_encoding != 0xff {
74            // Rust doesn't analyze exception types, so we don't care about the type table
75            reader.read_uleb128();
76        }
77
78        let call_site_encoding = Encoding::from(reader.read::<u8>()?);
79        let call_site_table_length = reader.read_uleb128()?;
80        let action_table = reader.current() + call_site_table_length;
81        let ip = frame.pc();
82        while reader.current() < action_table {
83            let cs_start = reader.read_with_encoding(call_site_encoding)?;
84            let cs_len = reader.read_with_encoding(call_site_encoding)?;
85            let cs_lpad = reader.read_with_encoding(call_site_encoding)?;
86            let cs_action = reader.read_uleb128()?;
87            // Callsite table is sorted by cs_start, so if we've passed the ip, we
88            // may stop searching.
89            if ip < fde.pc.start + cs_start {
90                break;
91            }
92            if ip < fde.pc.start + cs_start + cs_len {
93                if cs_lpad == 0 {
94                    return Some(EhAction::None);
95                } else {
96                    let lpad = lpad_base + cs_lpad;
97                    return Some(interpret_cs_action(cs_action, lpad));
98                }
99            }
100        }
101        // Ip is not present in the table.  This should not happen... but it does: issue
102        // #35011. So rather than returning EHAction::Terminate, we do this.
103        Some(EhAction::None)
104    } else {
105        Some(EhAction::None)
106    }
107}
108
109fn interpret_cs_action(cs_action: usize, lpad: usize) -> EhAction {
110    if cs_action == 0 {
111        // If cs_action is 0 then this is a cleanup (Drop::drop). We run these
112        // for both Rust panics and foreign exceptions.
113        EhAction::Cleanup(lpad)
114    } else {
115        // Stop unwinding Rust panics at catch_unwind.
116        EhAction::Catch(lpad)
117    }
118}

121#[unsafe(no_mangle)]
122#[unsafe(naked)]
123unsafe extern "C" fn _Unwind_Resume(_arg: usize) -> ! {
124    naked_asm!("jmp [rdi]")
125}
