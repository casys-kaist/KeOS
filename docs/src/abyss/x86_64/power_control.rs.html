7/// `machine_restart()` is emergency restart function since it lacks
8/// any synchorization technique.
9///
10/// This function exploits PS/2 I/O to touch the reset pin at the hardware, but
11/// if it fails, it tries to intentionally generate the triple fault by loading
12/// invalid IDT (Interrupt Descriptor Table) and invoke interrupt.
13///
14/// It's intended to be run at the end of safe restart or emergency situation.
15///
16/// This function should be run with interrupt disabled.
17pub unsafe fn restart() -> ! {
18    // Make sure that External Interrupt is disabled before we do the actual
19    // operation, regardless of actual CPU state.
20    unsafe {
21        asm!("cli");
22    }
23
24    let mut good = 0x02;
25    let port = Pio::new(0x64);
26    let mut ps2_try_cnt = 0;
27    while (good & 0x02) == 0x02 {
28        good = port.read_u8();
29        ps2_try_cnt += 1;
30        if ps2_try_cnt > 20 {
31            break;
32        }
33    }
34    if (good & 0x02) != 0x02 {
35        port.write_u8(0xFE);
36    }
37
38    for _ in 0..0x1000 {
39        core::hint::spin_loop();
40    }
41
42    // 8042 Restart failed! Try to triple fault
43    unsafe {
44        asm!("lidt [0]");
45        asm!("int3");
46        core::hint::unreachable_unchecked();
47    }
48}

50/// `machine_power_off()` is emergency shutdown function since it lacks
51/// any synchorization technique.
52///
53/// This function will try to put system in ACPI power down (S5), but if it
54/// fails, it will try to use emulator ports to power down the system.
55/// If every power down techniques are unavailable, it fall backs to halt.
56///
57/// It's intended to be run at the end of safe restart or emergency situation.
58///
59/// This function should be run with interrupt disabled.
60pub unsafe fn power_off() -> ! {
61    // Make sure that External Interrupt is disabled before we do the actual
62    // operation, regardless of actual CPU state.
63    unsafe {
64        asm!("cli");
65    }
66
67    // Trying QEMU specific power off method
68    let emulators = [
69        (Pio::new(0xB004), 0x2000),
70        (Pio::new(0x604), 0x2000),
71        (Pio::new(0x3004), 0x3400),
72        (Pio::new(0x600), 0x34),
73    ];
74    for emulator in emulators {
75        emulator.0.write_u16(emulator.1);
76    }
77
78    loop {
79        // Power off failed!
80        unsafe {
81            asm!("hlt");
82        }
83    }
84}
