1//! Kernel GS (General Segment)
2//!
3//! Used for internal data structure per CPU regardless of the running thread.
4
5use super::{intrinsics::cpuid, msr::Msr, segmentation::TSS, tss::TaskStateSegment};
6use crate::MAX_CPU;
7
8#[repr(C, align(1024))]
9#[derive(Clone, Debug)]
10pub struct KernelGS {
11    pub tss_addr: *const TaskStateSegment,
12    __user_rbp_backup: usize,
13    __r14_backup: usize,
14    __r15_backup: usize,
15    pub interrupt_frame: *const crate::interrupt::Registers,
16}
17
18const KERNEL_GS_INIT: KernelGS = KernelGS {
19    tss_addr: core::ptr::null(),
20    __user_rbp_backup: 0,
21    __r14_backup: 0,
22    __r15_backup: 0,
23    interrupt_frame: core::ptr::null(),
24};
25
26pub unsafe fn current() -> &'static mut KernelGS {
27    let ptr: *mut KernelGS;
28    unsafe {
29        ptr = Msr::<0xC000_0101>::read() as usize as *mut _;
30        &mut *ptr as &mut KernelGS
31    }
32}
33
34#[unsafe(no_mangle)]
35pub static mut KERNEL_GS_BASES: [KernelGS; MAX_CPU] = [KERNEL_GS_INIT; MAX_CPU];
36
37impl KernelGS {
38    pub fn new() -> Self {
39        let mut init = KERNEL_GS_INIT.clone();
40        init.tss_addr = unsafe { &TSS[cpuid()].0 as *const TaskStateSegment };
41
42        init
43    }
44
45    pub unsafe fn apply(self) {
46        unsafe {
47            KERNEL_GS_BASES[cpuid()] = self;
48            Msr::<0xC000_0101>::write(&mut KERNEL_GS_BASES[cpuid()] as *mut KernelGS as u64);
49        }
50    }
51}
52
53impl Default for KernelGS {
54    fn default() -> Self {
55        Self::new()
56    }
57}
