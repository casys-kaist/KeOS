16pub trait MmioHandler
17where
18    Self: Send + Sync,
19{
20    fn region(&self) -> MmioRegion;
21    fn handle(
22        &mut self,
23        p: &dyn Probe,
24        info: MmioInfo,
25        generic_vcpu_state: &mut GenericVCpuState,
26    ) -> Result<VmexitResult, VmError>;
27}

29/// Representation of interval.
30///
31/// This implements interval tree by overloading PartialOrd, Ord, PartialEq, and
32/// Eq with ordered map (BTreeMap).
33#[derive(Eq, Clone, Copy)]
34pub struct MmioRegion {
35    // (start, end]
36    pub start: Gpa,
37    pub end: Gpa,
38}
39
40impl core::fmt::Debug for MmioRegion {
41    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> Result<(), core::fmt::Error> {
42        write!(f, "MmioRegion({:?}, {:?}]", self.start, self.end)
43    }
44}
45
46impl core::cmp::Ord for MmioRegion {
47    fn cmp(&self, other: &Self) -> Ordering {
48        let has_overlapping = self.start < other.end && self.end > other.start;
49        if has_overlapping {
50            Ordering::Equal
51        } else {
52            self.start.cmp(&other.start)
53        }
54    }
55}
56
57impl core::cmp::PartialOrd for MmioRegion {
58    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
59        Some(self.cmp(other))
60    }
61}
62
63impl core::cmp::PartialEq for MmioRegion {
64    fn eq(&self, other: &Self) -> bool {
65        self.cmp(other) == Ordering::Equal
66    }
67}
68
69impl MmioRegion {
70    /// Create a new MmioRegion.
71    #[inline(always)]
72    pub fn new(gpa: Gpa, size: usize) -> Self {
73        Self {
74            start: gpa,
75            end: gpa + size,
76        }
77    }
78}

80/// Mmio vmexit controller.
81#[derive(Default)]
82pub struct Controller {
83    inner: BTreeMap<MmioRegion, Box<dyn MmioHandler>>,
84}
85
86#[derive(Debug)]
87pub enum Direction {
88    Write8 { dst: Gpa, src: u8 },
89    Write16 { dst: Gpa, src: u16 },
90    Write32 { dst: Gpa, src: u32 },
91    Write64 { dst: Gpa, src: u64 },
92}
93
94impl Direction {
95    fn get_mmio_addr(&self) -> Gpa {
96        match self {
97            Self::Write8 { dst: v, .. }
98            | Self::Write16 { dst: v, .. }
99            | Self::Write32 { dst: v, .. }
100            | Self::Write64 { dst: v, .. } => *v,
101        }
102    }
103}
104
105#[derive(Debug)]
106pub struct MmioInfo {
107    pub size: usize,
108    pub direction: Direction,
109}
110
111impl Controller {
112    /// Create a new mmio vmexit controller.
113    pub fn new() -> Self {
114        Controller {
115            inner: BTreeMap::new(),
116        }
117    }
118    /// Add a mmio region to the controller.
119    pub fn register(&mut self, p: impl MmioHandler + 'static) {
120        match self.inner.entry(p.region()) {
121            Entry::Occupied(_) => panic!("overwrapping mmio region"),
122            Entry::Vacant(v) => {
123                v.insert(Box::new(p));
124            }
125        }
126    }
127}
128
129impl kev::vmexits::VmexitController for Controller {
130    fn handle<P: Probe>(
131        &mut self,
132        reason: ExitReason,
133        p: &mut P,
134        generic_vcpu_state: &mut GenericVCpuState,
135    ) -> Result<VmexitResult, VmError> {
136        match reason.get_basic_reason() {
137            BasicExitReason::EptViolation {
138                qualification,
139                fault_addr,
140            } => {
141                assert!(
142                    qualification.contains(EptViolationQualification::BIT1),
143                    "rip: {:x}, {qualification:?}, {fault_addr:?}",
144                    generic_vcpu_state.vmcs.read(Field::GuestRip).unwrap()
145                );
146
147                let mmio_info = get_mmio_info(
148                    generic_vcpu_state.gprs,
149                    &generic_vcpu_state.vmcs.get_instruction(p)?,
150                    fault_addr.ok_or(VmError::HandleVmexitFailed(reason))?,
151                    *qualification,
152                );
153                if let Some(handler) = self.inner.get_mut(&MmioRegion::new(
154                    mmio_info.direction.get_mmio_addr(),
155                    mmio_info.size,
156                )) {
157                    handler
158                        .handle(p, mmio_info, generic_vcpu_state)
159                        .and_then(|r| generic_vcpu_state.vmcs.forward_rip().map(|_| r))
160                } else {
161                    Err(VmError::HandleVmexitFailed(reason))
162                }
163            }
164            _ => Err(VmError::HandleVmexitFailed(reason)),
165        }
166    }
167}

169fn get_mmio_value(insn: &Instruction, gprs: &GeneralPurposeRegisters) -> u64 {
170    match insn.op1_kind() {
171        OpKind::Register => match insn.op1_register() {
172            Register::AL => gprs.rax as u8 as u64,
173            Register::BL => gprs.rbx as u8 as u64,
174            Register::CL => gprs.rcx as u8 as u64,
175            Register::DL => gprs.rdx as u8 as u64,
176            Register::AX => gprs.rax as u16 as u64,
177            Register::BX => gprs.rbx as u16 as u64,
178            Register::CX => gprs.rcx as u16 as u64,
179            Register::DX => gprs.rdx as u16 as u64,
180            Register::EAX => gprs.rax as u32 as u64,
181            Register::EBX => gprs.rbx as u32 as u64,
182            Register::ECX => gprs.rcx as u32 as u64,
183            Register::EDX => gprs.rdx as u32 as u64,
184            Register::RAX => gprs.rax as u64,
185            Register::RBX => gprs.rbx as u64,
186            Register::RCX => gprs.rcx as u64,
187            Register::RDX => gprs.rdx as u64,
188            Register::R8L => gprs.r8 as u8 as u64,
189            Register::R9L => gprs.r9 as u8 as u64,
190            Register::R10L => gprs.r10 as u8 as u64,
191            Register::R11L => gprs.r11 as u8 as u64,
192            Register::R12L => gprs.r12 as u8 as u64,
193            Register::R13L => gprs.r13 as u8 as u64,
194            Register::R14L => gprs.r14 as u8 as u64,
195            Register::R15L => gprs.r15 as u8 as u64,
196            Register::R8W => gprs.r8 as u16 as u64,
197            Register::R9W => gprs.r9 as u16 as u64,
198            Register::R10W => gprs.r10 as u16 as u64,
199            Register::R11W => gprs.r11 as u16 as u64,
200            Register::R12W => gprs.r12 as u16 as u64,
201            Register::R13W => gprs.r13 as u16 as u64,
202            Register::R14W => gprs.r14 as u16 as u64,
203            Register::R15W => gprs.r15 as u16 as u64,
204            Register::R8D => gprs.r8 as u32 as u64,
205            Register::R9D => gprs.r9 as u32 as u64,
206            Register::R10D => gprs.r10 as u32 as u64,
207            Register::R11D => gprs.r11 as u32 as u64,
208            Register::R12D => gprs.r12 as u32 as u64,
209            Register::R13D => gprs.r13 as u32 as u64,
210            Register::R14D => gprs.r14 as u32 as u64,
211            Register::R15D => gprs.r15 as u32 as u64,
212            Register::R8 => gprs.r8 as u64,
213            Register::R9 => gprs.r9 as u64,
214            Register::R10 => gprs.r10 as u64,
215            Register::R11 => gprs.r11 as u64,
216            Register::R12 => gprs.r12 as u64,
217            Register::R13 => gprs.r13 as u64,
218            Register::R14 => gprs.r14 as u64,
219            Register::R15 => gprs.r15 as u64,
220            e => todo!("{e:?}"),
221        },
222        OpKind::Immediate8 => insn.immediate8() as u64,
223        OpKind::Immediate16 => insn.immediate16() as u64,
224        OpKind::Immediate32 => insn.immediate32() as u64,
225        OpKind::Immediate64 => insn.immediate64(),
226        OpKind::Immediate8to16 => insn.immediate8() as u64,
227        OpKind::Immediate8to32 => insn.immediate16() as u64,
228        OpKind::Immediate8to64 => insn.immediate16() as u64,
229        OpKind::Immediate32to64 => insn.immediate32() as u64,
230        e => unreachable!("{e:?}"),
231    }
232}

234fn get_mmio_info(
235    gprs: &mut GeneralPurposeRegisters,
236    insn: &Instruction,
237    fault_addr: Gpa,
238    _qualification: EptViolationQualification,
239) -> MmioInfo {
240    // Must be write access.
241    assert!(insn.op0_kind() == OpKind::Memory);
242
243    let size = insn.memory_size();
244
245    let direction = match size {
246        iced_x86::MemorySize::UInt8 => Direction::Write8 {
247            dst: fault_addr,
248            src: get_mmio_value(insn, gprs) as u8,
249        },
250        iced_x86::MemorySize::UInt16 => Direction::Write16 {
251            dst: fault_addr,
252            src: get_mmio_value(insn, gprs) as u16,
253        },
254        iced_x86::MemorySize::UInt32 => Direction::Write32 {
255            dst: fault_addr,
256            src: get_mmio_value(insn, gprs) as u32,
257        },
258        iced_x86::MemorySize::UInt64 => Direction::Write64 {
259            dst: fault_addr,
260            src: get_mmio_value(insn, gprs),
261        },
262        _ => unreachable!(),
263    };
264
265    let size: usize = match size {
266        iced_x86::MemorySize::UInt8 => 8,
267        iced_x86::MemorySize::UInt16 => 16,
268        iced_x86::MemorySize::UInt32 => 32,
269        iced_x86::MemorySize::UInt64 => 64,
270        _ => unreachable!(),
271    };
272    MmioInfo { size, direction }
