43/// Error for setup_vbsp.

51impl kev::vm::VmState for SimpleEptVmState {
52    type Error = Error;
53    type VcpuState = SimpleEptVcpuState;
54
55    fn vcpu_state(&self) -> Self::VcpuState {
56        let mut mmio_controller = mmio::Controller::new();
57        mmio_controller.register(PrinterDev::default());
58        SimpleEptVcpuState {
59            ept: ExtendedPageTable::new(),
60            page_table: PageTable(PageTableRoot::new_boxed()),
61            vmexit_controller: (hypercall::Controller::new(HypercallCtx), (mmio_controller)),
62        }
63    }
64
65    fn setup_vbsp(
66        &self,
67        vbsp_generic_state: &mut GenericVCpuState,
68        vbsp_vcpu_state: &mut Self::VcpuState,
69    ) -> Result<(), Self::Error> {
70        // Prepare code pages.
71        let mut pgs = keos::mm::ContigPages::new((self.code.len() + 0xfff) & !0xfff)
72            .expect("Failed to allocate code page")
73            .split();
74        // Copy code to page
75        let ((srcs, remainder), mut dsts) = (self.code.as_chunks::<0x1000>(), pgs.iter_mut());
76        for src in srcs {
77            let dst = dsts.next().unwrap();
78            dst.inner_mut().copy_from_slice(src);
79        }
80        let dst = dsts.next().unwrap();
81        dst.inner_mut()[..remainder.len()].copy_from_slice(remainder);
82
83        // Map into page table.
84        let mut base = pgs.first().unwrap().pa().into_usize();
85        vbsp_generic_state
86            .vmcs
87            .write(Field::GuestRip, base as u64)
88            .expect("Failed to write to guest rip.");
89
90        for pg in pgs.into_iter() {
91            vbsp_vcpu_state
92                .page_table
93                .map(
94                    Va::new(base).unwrap(),
95                    pg,
96                    Permission::READ | Permission::EXECUTABLE,
97                )
98                .map_err(Error::PageTableError)?;
99            base += 0x1000;
100        }
101        let gdt = SystemTableRegister::new(unsafe { (&raw const SEGMENT_TABLE).as_ref() }.unwrap());
102        let base = gdt.address & !(PAGE_MASK as u64);
103        for kva in (base..gdt.address + gdt.size as u64).step_by(0x1000) {
104            let kva = Kva::new(kva as usize).unwrap();
105            unsafe {
106                vbsp_vcpu_state
107                    .page_table
108                    .do_map(kva.into_va(), kva.into_pa(), Permission::READ)
109                    .map_err(Error::PageTableError)?;
110            }
111        }
112
113        // Add mmio area.
114        unsafe {
115            vbsp_vcpu_state
116                .page_table
117                .do_map(
118                    Va::new(0xcafe0000).unwrap(),
119                    Pa::new(0xcafe0000).unwrap(),
120                    Permission::READ | Permission::WRITE | Permission::EXECUTABLE,
121                )
122                .map_err(Error::PageTableError)?;
123        }
124        vbsp_vcpu_state
125            .ept
126            .map(
127                Gpa::new(0xcafe0000).unwrap(),
128                Page::new(),
129                EptPermission::READ,
130            )
131            .map_err(Error::EptError)?;
132        // gpa -> hpa mappings.
133        unsafe {
134            use core::slice::from_raw_parts;
135            let table = vbsp_vcpu_state
136                .add_gpa_mapping(vbsp_vcpu_state.page_table.pa().into_usize() & !PAGE_MASK)
137                .map_err(Error::EptError)?;
138
139            for pml4e in from_raw_parts(table, 512).iter().filter(|e| *e & 1 != 0) {
140                let ntable = vbsp_vcpu_state
141                    .add_gpa_mapping(Pml4e(*pml4e).pa().unwrap().into_usize())
142                    .map_err(Error::EptError)?;
143                for pdpe in from_raw_parts(ntable, 512).iter().filter(|e| *e & 1 != 0) {
144                    let ntable = vbsp_vcpu_state
145                        .add_gpa_mapping(Pdpe(*pdpe).pa().unwrap().into_usize())
146                        .map_err(Error::EptError)?;
147                    for pde in from_raw_parts(ntable, 512).iter().filter(|e| *e & 1 != 0) {
148                        let ntable = vbsp_vcpu_state
149                            .add_gpa_mapping(Pde(*pde).pa().unwrap().into_usize())
150                            .map_err(Error::EptError)?;
151                        for pte in from_raw_parts(ntable, 512).iter().filter(|e| *e & 1 != 0) {
152                            let pa = Pte(*pte).pa().unwrap().into_usize();
153                            if pa != 0xcafe0000 {
154                                vbsp_vcpu_state
155                                    .add_gpa_mapping(pa)
156                                    .map_err(Error::EptError)?;
157                            }
158                        }
159                    }
160                }
161            }
162        }
163        // Run a guest on 64bit mode directly.
164        let vmcs = &vbsp_generic_state.vmcs;
165        vmcs.write(
166            Field::GuestCsSelector,
167            Segment::KernelCode.into_selector().pack() as u64,
168        )
169        .map_err(Error::VmError)?;
170        vmcs.write(
171            Field::GuestEsSelector,
172            Segment::KernelData.into_selector().pack() as u64,
173        )
174        .map_err(Error::VmError)?;
175        vmcs.write(
176            Field::GuestSsSelector,
177            Segment::KernelData.into_selector().pack() as u64,
178        )
179        .map_err(Error::VmError)?;
180        vmcs.write(
181            Field::GuestDsSelector,
182            Segment::KernelData.into_selector().pack() as u64,
183        )
184        .map_err(Error::VmError)?;
185        vmcs.write(
186            Field::GuestFsSelector,
187            Segment::KernelData.into_selector().pack() as u64,
188        )
189        .map_err(Error::VmError)?;
190        vmcs.write(
191            Field::GuestGsSelector,
192            Segment::KernelData.into_selector().pack() as u64,
193        )
194        .map_err(Error::VmError)?;
195        vmcs.write(
196            Field::GuestTrSelector,
197            Segment::KernelData.into_selector().pack() as u64,
198        )
199        .map_err(Error::VmError)?;
200        vmcs.write(Field::GuestLdtrSelector, 0)
201            .map_err(Error::VmError)?;
202
203        vmcs.write(Field::GuestCsBase, 0).map_err(Error::VmError)?;
204        vmcs.write(Field::GuestEsBase, 0).map_err(Error::VmError)?;
205        vmcs.write(Field::GuestSsBase, 0).map_err(Error::VmError)?;
206        vmcs.write(Field::GuestDsBase, 0).map_err(Error::VmError)?;
207        vmcs.write(Field::GuestFsBase, 0).map_err(Error::VmError)?;
208        vmcs.write(Field::GuestGsBase, 0).map_err(Error::VmError)?;
209        vmcs.write(Field::GuestLdtrBase, 0)
210            .map_err(Error::VmError)?;
211        vmcs.write(
212            Field::GuestGdtrBase,
213            SystemTableRegister::new(unsafe { (&raw const SEGMENT_TABLE).as_ref() }.unwrap())
214                .address,
215        )
216        .map_err(Error::VmError)?;
217        vmcs.write(Field::GuestIdtrBase, 0)
218            .map_err(Error::VmError)?;
219        vmcs.write(Field::GuestTrBase, 0).map_err(Error::VmError)?;
220
221        vmcs.write(Field::GuestCsLimit, u64::MAX)
222            .map_err(Error::VmError)?;
223        vmcs.write(Field::GuestEsLimit, u64::MAX)
224            .map_err(Error::VmError)?;
225        vmcs.write(Field::GuestSsLimit, u64::MAX)
226            .map_err(Error::VmError)?;
227        vmcs.write(Field::GuestDsLimit, u64::MAX)
228            .map_err(Error::VmError)?;
229        vmcs.write(Field::GuestFsLimit, u64::MAX)
230            .map_err(Error::VmError)?;
231        vmcs.write(Field::GuestGsLimit, u64::MAX)
232            .map_err(Error::VmError)?;
233        vmcs.write(Field::GuestLdtrLimit, 0xffff)
234            .map_err(Error::VmError)?;
235        vmcs.write(Field::GuestGdtrLimit, 0xffff)
236            .map_err(Error::VmError)?;
237        vmcs.write(Field::GuestIdtrLimit, 0xffff)
238            .map_err(Error::VmError)?;
239        vmcs.write(Field::GuestTrLimit, 0x67)
240            .map_err(Error::VmError)?;
241
242        vmcs.write(Field::GuestCsAccessRights, 0xa09b)
243            .map_err(Error::VmError)?;
244        vmcs.write(Field::GuestEsAccessRights, 0xc093)
245            .map_err(Error::VmError)?;
246        vmcs.write(Field::GuestSsAccessRights, 0xc093)
247            .map_err(Error::VmError)?;
248        vmcs.write(Field::GuestDsAccessRights, 0xc093)
249            .map_err(Error::VmError)?;
250        vmcs.write(Field::GuestFsAccessRights, 0xc093)
251            .map_err(Error::VmError)?;
252        vmcs.write(Field::GuestGsAccessRights, 0xc093)
253            .map_err(Error::VmError)?;
254        vmcs.write(Field::GuestTrAccessRights, 0x8b)
255            .map_err(Error::VmError)?;
256        vmcs.write(Field::GuestLdtrAccessRights, 0x10000)
257            .map_err(Error::VmError)?;
258
259        vmcs.write(Field::GuestActivityState, 0)
260            .map_err(Error::VmError)?;
261        vmcs.write(Field::GuestInterruptibilityState, 0)
262            .map_err(Error::VmError)?;
263
264        vmcs.write(Field::GuestCr0, Cr0::current().bits())
265            .map_err(Error::VmError)?;
266        vmcs.write(Field::GuestCr4, Cr4::current().bits())
267            .map_err(Error::VmError)?;
268        vmcs.write(Field::GuestIa32Efer, 0x500)
269            .map_err(Error::VmError)?;
270
271        vmcs.write(Field::GuestLinkPointer, 0xffff_ffff)
272            .map_err(Error::VmError)?;
273        vmcs.write(Field::GuestLinkPointerHi, 0xffff_ffff)
274            .map_err(Error::VmError)?;
275
276        vmcs.write(Field::GuestDr7, 0).map_err(Error::VmError)?;
277        vmcs.write(Field::GuestIa32Debugctl, 0)
278            .map_err(Error::VmError)?;
279
280        vmcs.write(Field::GuestRflags, Rflags::_1.bits())
281            .map_err(Error::VmError)?;
282
283        vmcs.write(
284            Field::GuestCr3,
285            vbsp_vcpu_state.page_table.pa().into_usize() as u64,
286        )
287        .map_err(Error::VmError)?;
288        Ok(())
289    }

292/// The Vcpu state of NoEptVmState.
293pub struct SimpleEptVcpuState {
294    ept: ExtendedPageTable,
295    page_table: PageTable,
296    vmexit_controller: (hypercall::Controller<HypercallCtx>, mmio::Controller),
297}
298
299impl SimpleEptVcpuState {
300    unsafe fn add_gpa_mapping(&mut self, pa: usize) -> Result<*const usize, EptMappingError> {
301        unsafe {
302            let hpa = Pa::new(pa).unwrap();
303            self.ept
304                .do_map(
305                    Gpa::new(pa).unwrap(),
306                    hpa,
307                    EptPermission::READ | EptPermission::WRITE | EptPermission::EXECUTABLE,
308                )
309                .map(|_| hpa.into_kva().into_usize() as *const usize)
310        }
311    }
312}
313
314impl kev::vcpu::VCpuState for SimpleEptVcpuState {
315    fn pinbase_ctls(&self) -> VmcsPinBasedVmexecCtl {
316        VmcsPinBasedVmexecCtl::EXTERNAL_INTERRUPT_EXITING
317    }
318    fn procbase_ctls(&self) -> VmcsProcBasedVmexecCtl {
319        VmcsProcBasedVmexecCtl::HLT_EXITING
320            | VmcsProcBasedVmexecCtl::CR3LOADEXIT
321            | VmcsProcBasedVmexecCtl::UNCONDIOEXIT
322    }
323    fn procbase_ctls2(&self) -> VmcsProcBasedSecondaryVmexecCtl {
324        VmcsProcBasedSecondaryVmexecCtl::ENABLE_RDTSCP | VmcsProcBasedSecondaryVmexecCtl::ENABLE_EPT
325    }
326    fn entry_ctls(&self) -> VmcsEntryCtl {
327        VmcsEntryCtl::IA32E_MODE_GUEST | VmcsEntryCtl::LOAD_IA32_EFER
328    }
329    fn exit_ctls(&self) -> VmcsExitCtl {
330        VmcsExitCtl::HOST_ADDRESS_SPACE_SIZE | VmcsExitCtl::ACK_INTR_ON_EXIT
331    }
332    fn init_guest_state(&self, vmcs: &ActiveVmcs) -> Result<(), VmError> {
333        vmcs.write(
334            Field::Eptptr,
335            self.ept.pa().into_usize() as u64 | (3 << 3) | 6,
336        )?;
337        Ok(())
338    }
339
340    fn handle_vmexit(
341        &mut self,
342        generic_vcpu_state: &mut GenericVCpuState,
343    ) -> Result<VmexitResult, VmError> {
344        let exit_reason = generic_vcpu_state.vmcs.exit_reason()?;
345        let Self {
346            ept: mem,
347            vmexit_controller,
348            ..
349        } = self;
350        vmexit_controller.handle(exit_reason, mem, generic_vcpu_state)
351    }
