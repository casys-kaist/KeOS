15/// Virtual Machine Control State.
16///
17/// This holds virtual machine specific information.
18///
19/// ## Details
20/// See Intel® 64 and IA-32 Architectures Software Developer’s Manual, 24.2
21/// FORTMAT OF VMCS.
22#[repr(align(4096))]
23pub struct Vmcs {
24    /// Bits 30:0: VMCS revision identifier
25    /// Bit 31: shadow-VMCS indicator (see Section 24.10)
26    rev_id: u32,
27    /// VMX-abort indicator
28    indicator: u32,
29    /// VMCS data (implementation-specific format)
30    _data: [u8; 0x1000 - 4],
31}

49    VmresumeWithNonlaunchedVmcs,
50    /// VMRESUME after VMXOFF
51    VmresumeAfterVmxoff,
52    /// VM entry with invalid control field(s)
53    InvalidCs,
54    /// VM entry with invalid host-state field(s)2
55    InvalidHostState,
56    /// VMPTRLD with invalid physical address
57    VmPtrLdWithInvAddr,
58    /// VMPTRLD with VMXON pointer
59    VmPtrLdWithVmxOn,
60    /// VMPTRLD with incorrect VMCS revision identifier
61    VmPtrLdWithIncorrectRevId,

63    UnsupportedVmcsField,
64    /// VMWRITE to read-only VMCS component

67    VmxonInVmxRoot,

76    /// treatment of SMIs and SMMk)
77    VmcallWithNonclearVmcs,
78    /// VMCALL with invalid VM-exit control fields
79    VmcallWithInvVmexitCs,
80    /// VMCALL with incorrect MSEG revision identifier (when attempting to
81    /// activate the dual-monitor treatment of SMIs and SMM)
82    VmcallWithIncorrectMsegRevId,
83    /// VMXOFF under dual-monitor treatment of SMIs and SMM
84    VmxoffUnderDualMontiroTreatment,
85    /// VMCALL with invalid SMM-monitor features (when attempting to activate
86    /// the dual-monitor treatment of SMIs and SMM)
87    VmcallWithInvSmmMonitor,
88    /// VM entry with invalid VM-execution control fields in executive VMCS

91    /// VM entry with events blocked by MOV SS.
92    VmentryWithEventBlockedByMovss,
93    /// Invalid operand to INVEPT/INVVPID.

299    /// Create a new Vmcs struct.
300    pub fn new() -> Self {
301        // Load vmx realated msrs.
302        let vmx_basic = Msr::<IA32_VMX_BASIC>::read();
303        let (rev_id, _vmcs_size) = (vmx_basic as u32, (vmx_basic >> 32) & 0xfff);
304
305        Self {
306            rev_id,
307            indicator: 0,
308            _data: [0; 0x1000 - 4],
309        }
310    }
311
312    pub(crate) fn on(&self) -> Result<(), InstructionError> {
313        unsafe {
314            let err: i8;
315            let pa = Kva::new(self as *const _ as usize)
316                .unwrap()
317                .into_pa()
318                .into_usize();
319            asm!(
320                "clc",
321                "vmxon [{}]",
322                "setna {}",
323                in(reg) &pa,
324                out(reg_byte) err
325            );
326            if err != 0 {
327                Err(Self::instruction_error())
328            } else {
329                Ok(())
330            }
331        }
332    }

378    pub(crate) fn instruction_error() -> InstructionError {
379        unsafe {
380            let err: i8;
381            let v: u64;
382            asm!(
383                "clc",
384                "vmread {}, {}",
385                "setna {}",
386                out(reg) v,
387                in(reg) Field::InstructionError as u64,
388                out(reg_byte) err
389            );
390            if err != 0 {
391                InstructionError::Unknown
392            } else {
393                match v {
394                    1 => InstructionError::VmcallInVmxRoot,
395                    2 => InstructionError::VmclearWithInvAddr,
396                    3 => InstructionError::VmclearWithVmxon,
397                    4 => InstructionError::VmresumeWithNonclearVmcs,
398                    5 => InstructionError::VmresumeWithNonlaunchedVmcs,
399                    6 => InstructionError::VmresumeAfterVmxoff,
400                    7 => InstructionError::InvalidCs,
401                    8 => InstructionError::InvalidHostState,
402                    9 => InstructionError::VmPtrLdWithInvAddr,
403                    10 => InstructionError::VmPtrLdWithVmxOn,
404                    11 => InstructionError::VmPtrLdWithIncorrectRevId,
405                    12 => InstructionError::UnsupportedVmcsField,
406                    13 => InstructionError::WriteToRoField,
407                    15 => InstructionError::VmxonInVmxRoot,
408                    16 => InstructionError::VmEntryWithInvalidExecVmcs2,
409                    17 => InstructionError::VmEntryWithNonlaucnhedExecVmcs2,
410                    18 => InstructionError::VmEntryWithExecVmcs,
411                    19 => InstructionError::VmcallWithNonclearVmcs,
412                    20 => InstructionError::VmcallWithInvVmexitCs,
413                    22 => InstructionError::VmcallWithIncorrectMsegRevId,
414                    23 => InstructionError::VmxoffUnderDualMontiroTreatment,
415                    24 => InstructionError::VmcallWithInvSmmMonitor,
416                    25 => InstructionError::VmentryWithInvVmExecCs,
417                    26 => InstructionError::VmentryWithEventBlockedByMovss,
418                    28 => InstructionError::InvalidOperandToInveptInvvpid,
419                    _ => InstructionError::Unknown,
420                }
421            }
422        }
423    }

561    fn parse_basic_reason(&self, reason: u64) -> Result<BasicExitReason, VmError> {
562        Ok(match reason {
563            0x0 => BasicExitReason::ExceptionOrNmi,
564            0x1 => {
565                // 24.9.2 Information for VM Exits Due to Vectored Events
566                let info = self.read(Field::VmexitInterruptionInfo)? as u32;
567                BasicExitReason::ExternalInt(if info & 0x8000_0000 != 0 {
568                    Some(ExternalIntInfo {
569                        host_int: info as u8,
570                        interruption_type: match (info >> 8) & 7 {
571                            0 => InterruptionType::ExternalInt,
572                            2 => InterruptionType::Nmi,
573                            3 => InterruptionType::HardwareException,
574                            5 => InterruptionType::PrivSoftwareException,
575                            6 => InterruptionType::SoftwareExeception,
576                            _ => unreachable!(),
577                        },
578                        error_code_valid: info & (1 << 11) != 0,
579                        nmi_unblocked_by_iret: info & (1 << 12) != 0,
580                    })
581                } else {
582                    None
583                })
584            }
585            0x2 => BasicExitReason::TripleFault,
586            0x3 => BasicExitReason::InitSignal,
587            0x4 => BasicExitReason::StartupIpi,
588            0x5 => BasicExitReason::IoSmi,
589            0x6 => BasicExitReason::OtherSmi,
590            0x7 => BasicExitReason::InterruptWindow,
591            0x9 => BasicExitReason::TaskSwitch,
592            0xA => BasicExitReason::Cpuid,
593            0xC => BasicExitReason::Hlt,
594            0xD => BasicExitReason::Invd,
595            0xE => BasicExitReason::Invlpg,
596            0xF => BasicExitReason::Rdpmc,
597            0x10 => BasicExitReason::Rdtsc,
598            0x11 => BasicExitReason::Rsm,
599            0x12 => BasicExitReason::Vmcall,
600            0x13 => BasicExitReason::Vmclear,
601            0x14 => BasicExitReason::Vmlaunch,
602            0x15 => BasicExitReason::Vmptrld,
603            0x16 => BasicExitReason::Vmptrst,
604            0x17 => BasicExitReason::Vmread,
605            0x18 => BasicExitReason::Vmresume,
606            0x19 => BasicExitReason::Vmwrite,
607            0x1A => BasicExitReason::Vmxoff,
608            0x1B => BasicExitReason::Vmxon,
609            0x1C => BasicExitReason::MovCr,
610            0x1D => BasicExitReason::MovDr,
611            0x1E => BasicExitReason::IoInstruction,
612            0x1F => BasicExitReason::Rdmsr,
613            0x20 => BasicExitReason::Wrmsr,
614            0x21 => BasicExitReason::EntfailGuestState,
615            0x22 => BasicExitReason::EntfailMsrLoading,
616            0x24 => BasicExitReason::Mwait,
617            0x25 => BasicExitReason::Mtf,
618            0x27 => BasicExitReason::Monitor,
619            0x28 => BasicExitReason::Pause,
620            0x29 => BasicExitReason::EntfailMachineChk,
621            0x2B => BasicExitReason::TprBelowThreshold,
622            0x2C => BasicExitReason::ApicAccess,
623            0x2E => BasicExitReason::AccessGdtrOrIdtr,
624            0x2F => BasicExitReason::AccessLdtrOrTr,

626                qualification: EptViolationQualification::from_bits_truncate(
627                    self.read(Field::VmexitQualification)?,
628                ),
629                fault_addr: Gpa::new(self.read(Field::GuestPhysicalAddr)? as usize),
630            },
631            0x31 => BasicExitReason::EptMisconfig,
632            0x32 => BasicExitReason::Invept,
633            0x33 => BasicExitReason::Rdtscp,
634            0x34 => BasicExitReason::VmxPreemptTimer,
635            0x35 => BasicExitReason::Invvpid,
636            0x36 => BasicExitReason::Wbinvd,
637            0x37 => BasicExitReason::Xsetbv,
638            _ => BasicExitReason::Unknown,
639        })
640    }
641
642    /// Resolve the exit reason of the activated vmcs.
643    pub fn exit_reason(&self) -> Result<ExitReason, VmError> {
644        let reason = self.read(Field::VmexitReason)?;
645        match reason {
646            reason if reason & 0x20000000 != 0 => Ok(ExitReason::ExitFromVmxRootOperation(
647                self.parse_basic_reason(reason & 0xffff)?,
648            )),
649            reason if reason & 0x80000000 != 0 => Ok(ExitReason::EntryFailure(
650                self.parse_basic_reason(reason & 0xffff)?,
651            )),
652            reason => Ok(ExitReason::BasicExitReason(
653                self.parse_basic_reason(reason)?,
654            )),
655        }
656    }
657
658    /// Get the instruction that rip pointed.
659    pub fn get_instruction<P: Probe>(&self, p: &P) -> Result<Instruction, VmError> {
660        let rip = self
661            .read(Field::GuestRip)
662            .map(|v| Gva::new(v as usize).expect("Invalid va for guest os."))?;
663        let len = self.read(Field::VmexitInstructionLength)? as usize;
664        assert!(len <= 11);
665        // Every intel instruction is at most 11 bytes.
666        let mut bytes = [0; 11];
667        // Pull to the buffer.
668        bytes[..len].copy_from_slice(unsafe {
669            core::slice::from_raw_parts(
670                p.gva2hva(self, rip)
671                    .ok_or_else(|| {
672                        VmError::ControllerError(Box::new(format!(
673                            "Invalid memory access at {rip:?}",
674                        )))
675                    })?
676                    .into_usize() as *const u8,
677                len,
678            )
679        });
680
681        let mut decoder = Decoder::with_ip(64, &bytes, 0, DecoderOptions::NONE);
682        let mut insn = Instruction::default();
683        if decoder.can_decode() {
684            decoder.decode_out(&mut insn);
685            Ok(insn)
686        } else {
687            Err(VmError::FailedToDecodeInstruction)
688        }
689    }
690
691    /// Forward to the next instruction.
692    pub fn forward_rip(&self) -> Result<(), VmError> {
693        self.write(
694            Field::GuestRip,
695            self.read(Field::GuestRip)? + self.read(Field::VmexitInstructionLength)?,
696        )
697    }

793bitflags::bitflags! {
794    /// Exit Qualification for EPT Violations
795    ///
796    /// See Intel Manual volume 3C. Table 28-7. Exit Qualification for EPT Violations
797    pub struct EptViolationQualification: u64 {
798        /// Set if the access causing the EPT violation was a data read.
799        const BIT0 = 1 << 0;
800        ///  Set if the access causing the EPT violation was a data write.
801        const BIT1 = 1 << 1;
802        /// Set if the access causing the EPT violation was an instruction fetch.
803        const BIT2 = 1 << 2;
804        /// The logical-AND of bit 0 in the EPT paging-structure entries used to translate the guest-physical address of the
805        /// access causing the EPT violation (indicates whether the guest-physical address was readable).
806        const BIT3 = 1 << 3;
807        /// The logical-AND of bit 1 in the EPT paging-structure entries used to translate the guest-physical address of the
808        /// access causing the EPT violation (indicates whether the guest-physical address was writeable).
809        const BIT4 = 1 << 4;
810        /// The logical-AND of bit 2 in the EPT paging-structure entries used to translate the guest-physical address of the
811        /// access causing the EPT violation.
812        /// If the “mode-based execute control for EPT” VM-execution control is 0, this indicates whether the guest-physical
813        /// address was executable. If that control is 1, this indicates whether the guest-physical address was executable
814        /// for supervisor-mode linear addresses.
815        const BIT5 = 1 << 5;
816        /// If the “mode-based execute control” VM-execution control is 0, the value of this bit is undefined. If that control is
817        /// 1, this bit is the logical-AND of bit 10 in the EPT paging-structure entries used to translate the guest-physical
818        /// address of the access causing the EPT violation. In this case, it indicates whether the guest-physical address was
819        /// executable for user-mode linear addresses.
820        const BIT6 = 1 << 6;
821        /// Set if the guest linear-address field is valid.
822        /// The guest linear-address field is valid for all EPT violations except those resulting from an attempt to load the
823        /// guest PDPTEs as part of the execution of the MOV CR instruction and those due to trace-address pre-translation
824        /// (TAPT; Section 26.5.4).
825        const BIT7 = 1 << 7;
826        /// If bit 7 is 1:
827        /// • Set if the access causing the EPT violation is to a guest-physical address that is the translation of a linear
828        /// address.
829        /// • Clear if the access causing the EPT violation is to a paging-structure entry as part of a page walk or the
830        /// update of an accessed or dirty bit.
831        /// Reserved if bit 7 is 0 (cleared to 0)
832        const BIT8 = 1 << 8;
833        /// If bit 7 is 1, bit 8 is 1, and the processor supports advanced VM-exit information for EPT violations,3 this bit is 0
834        /// if the linear address is a supervisor-mode linear address and 1 if it is a user-mode linear address. (If CR0.PG = 0,
835        /// the translation of every linear address is a user-mode linear address and thus this bit will be 1.) Otherwise, this
836        /// bit is undefined.
837        const BIT9 = 1 << 9;
838        /// If bit 7 is 1, bit 8 is 1, and the processor supports advanced VM-exit information for EPT violations,3 this bit is 0
839        /// if paging translates the linear address to a read-only page and 1 if it translates to a read/write page. (If CR0.PG =
840        /// 0, every linear address is read/write and thus this bit will be 1.) Otherwise, this bit is undefined.
841        const BIT10 = 1 << 10;
842        /// If bit 7 is 1, bit 8 is 1, and the processor supports advanced VM-exit information for EPT violations,3 this bit is 0
843        /// if paging translates the linear address to an executable page and 1 if it translates to an execute-disable page. (If
844        /// CR0.PG = 0, CR4.PAE = 0, or IA32_EFER.NXE = 0, every linear address is executable and thus this bit will be 0.)
845        /// Otherwise, this bit is undefined.
846        const BIT11 = 1 << 11;
847        /// NMI unblocking due to IRET (see Section 28.2.3).
848        const BIT12 = 1 << 12;
849        /// Set if the access causing the EPT violation was a shadow-stack access.
850        const BIT13 = 1 << 13;
851        /// If supervisor shadow-stack control is enabled (by setting bit 7 of EPTP), this bit is the same as bit 60 in the EPT
852        /// paging-structure entry that maps the page of the guest-physical address of the access causing the EPT violation.
853        /// Otherwise (or if translation of the guest-physical address terminates before reaching an EPT paging-structure
854        /// entry that maps a page), this bit is undefined.
855        const BIT14 = 1 << 14;
856        /// This bit is set if the EPT violation was caused as a result of guest-paging verification. See Section 29.3.3.2.
857        const BIT15 = 1 << 15;
858        /// This bit is set if the access was asynchronous to instruction execution not the result of event delivery. The bit is
859        /// set if the access is related to trace output by Intel PT (see Section 26.5.4), accesses related to PEBS on
860        /// processors with the “EPT-friendly” enhancement (see Section 20.9.5), or to user-interrupt delivery (see Section
861        /// 7.4.2). Otherwise, this bit is cleared.
862        const BIT16 = 1 << 16;
863    }
864}
