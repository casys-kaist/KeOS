33bitflags::bitflags! {
34    /// Table 24-5. Definitions of Pin-Based VM-Execution Controls.
35    pub struct VmcsPinBasedVmexecCtl: u32 {
36        /// If this control is 1, external interrupts cause VM exits.
37        /// Otherwise, they are delivered normally through the guest interrupt-descriptor table (IDT).
38        /// If this control is 1, the value of RFLAGS. IF does not affect interrupt blocking.
39        const EXTERNAL_INTERRUPT_EXITING = 1 << 0;
40        /// If this control is 1, non-maskable interrupts (NMIs) cause VM exits.
41        /// Otherwise, they are delivered normally using descriptor 2 of the IDT.
42        /// This control also determines interactions between IRET and blocking by NMI (see Section 25.3).
43        const NMI_EXITING = 1 << 3;
44        /// If this control is 1, NMIs are never blocked and the “blocking by NMI” bit (bit 3) in the
45        /// interruptibility-state field indicates “virtual-NMI blocking” (see Table 24-3).
46        /// This control also interacts with the “NMI-window exiting” VM-execution control (see Section 24.6.2)
47        const VIRTUAL_NMIS = 1 << 5;
48        /// If this control is 1, the VMX-preemption timer counts down in VMX non-root operation;
49        /// see Section 25.5.1. A VM exit occurs when the timer counts down to zero; see Section 25.2.
50        const ACTIVE_VMX_PREEMPTION_TIMER = 1 << 6;
51        /// If this control is 1, the processor treats interrupts with the posted-interrupt notification vector
52        /// (see Section 24.6.8) specially, updating the virtual-APIC page with posted-interrupt requests (see Section 29.6).
53        const PROCESS_POSTED_INTERRUPT = 1 << 7;
54    }

58    /// Table 24-6. Definitions of Primary Processor-Based VM-Execution Controls.
59    pub struct VmcsProcBasedVmexecCtl: u32 {
60        /// If this control is 1, a VM exit occurs at the beginning of any instruction if RFLAGS.IF = 1 and
61        /// there are no other blocking of interrupts (see Section 24.4.2).
62        const INTRWINEXIT = 1 << 2;
63        /// This control determines whether executions of RDTSC, executions of RDTSCP, and executions
64        /// of RDMSR that read from the IA32_TIME_STAMP_COUNTER MSR return a value modified by
65        /// the TSC offset field (see Section 24.6.5 and Section 25.3).
66        const USETSCOFF	= 1 << 3;
67        /// This control determines whether executions of HLT cause VM exits.
68        const HLT_EXITING = 1 << 7;
69        /// This determines whether executions of INVLPG cause VM exits.
70        const INVLPGEXIT = 1 << 9;
71        /// This control determines whether executions of MWAIT cause VM exits
72        const MWAITEXIT	= 1 << 10;
73        /// This control determines whether executions of RDPMC cause VM exits.
74        const RDPMCEXIT	= 1 << 11;
75        /// This control determines whether executions of RDTSC and RDTSCP cause VM exits.
76        const RDTSCEXIT	= 1 << 12;
77        /// In conjunction with the CR3-target controls (see Section 24.6.7),
78        /// this control determines whether executions of MOV to CR3 cause VM exits. See Section 25.1.3.
79        ///
80        /// The first processors to support the
81        /// virtual-machine extensions supported only the 1-setting of this control.
82        const CR3LOADEXIT = 1 << 15;
83        /// This control determines whether executions of MOV from CR3 cause VM exits.
84        ///
85        /// The first processors to support the virtual-machine extensions supported only the 1-setting of this control.
86        const CR3STOREXIT = 1 << 16;
87        /// This control determines whether the tertiary processor-based VM-execution controls are used.
88        /// If this control is 0, the logical processor operates as if all the tertiary processor-based
89        /// VM-execution controls were also 0.
90        const ACTIVETETCTL = 1 << 17;
91        /// This control determines whether executions of MOV to CR8 cause VM exits.
92        const CR8LOADEXIT = 1 << 19;
93        /// This control determines whether executions of MOV from CR8 cause VM exits.
94        const CR8STOREEXIT = 1 << 20;
95        /// Setting this control to 1 enables TPR virtualization and other APIC-virtualization features. See Chapter 29.
96        const USETPRSHADOW = 1 << 21;
97        /// If this control is 1, a VM exit occurs at the beginning of any instruction if there is no virtual NMI blocking (see Section 24.4.2)
98        const NMIWINEXIT = 1 << 22;
99        /// This control determines whether executions of MOV DR cause VM exits.
100        const MOVDREXIT	= 1 << 23;
101        /// This control determines whether executions of I/O instructions
102        /// (IN, INS/INSB/INSW/INSD, OUT, and OUTS/OUTSB/OUTSW/OUTSD) cause VM exits.
103        const UNCONDIOEXIT = 1 << 24;
104        /// This control determines whether I/O bitmaps are used to restrict executions
105        /// of I/O instructions (see Section 24.6.4 and Section 25.1.3).
106        ///
107        /// For this control, “0” means “do not use I/O bitmaps” and “1” means “use I/O bitmaps.”
108        /// If the I/O bitmaps are used, the setting of the “unconditional I/O exiting” control is ignored.
109        const USEIOBMP = 1 << 25;
110        /// If this control is 1, the monitor trap flag debugging feature is enabled. See Section 25.5.2.
111        const MTF = 1 << 27;
112        /// This control determines whether MSR bitmaps are used to control execution of the RDMSR and
113        /// WRMSR instructions (see Section 24.6.9 and Section 25.1.3).
114        ///
115        /// For this control, “0” means “do not use MSR bitmaps” and “1” means “use MSR bitmaps.”
116        /// If the MSR bitmaps are not used, all executions of the RDMSR and WRMSR instructions cause VM exits.
117        const USEMSRBMP	= 1 << 28;
118        /// This control determines whether executions of MONITOR cause VM exits.
119        const MONITOREXIT = 1 << 29;
120        /// This control determines whether executions of PAUSE cause VM exits.
121        const PAUSEEXIT	= 1 << 30;
122        /// This control determines whether the secondary processor-based VM-execution controls are used.
123        /// If this control is 0, the logical processor operates as if all the secondary processor-based
124        /// VM-execution controls were also 0.
125        const ACTIVATE_SECONDARY_CTL = 1 << 31;
126    }

130    /// Table 24-7. Definitions of Secondary Processor-Based VM-Execution Controls.
131    pub struct VmcsProcBasedSecondaryVmexecCtl: u32 {
132        /// If this control is 1, the logical processor treats specially accesses to the page with the APICaccess address. See Section 29.4.
133        const VIRTUALIZE_APIC_ACCESSES = 1 << 0;
134        /// If this control is 1, extended page tables (EPT) are enabled. See Section 28.3.
135        const ENABLE_EPT = 1 << 1;
136        /// This control determines whether executions of LGDT, LIDT, LLDT, LTR, SGDT, SIDT, SLDT, and STR cause VM exits.
137        const DESCRIPTOR_TABLE_EXITING = 1 << 2;
138        /// If this control is 0, any execution of RDTSCP causes an invalid-opcode exception (#UD).
139        const ENABLE_RDTSCP = 1 << 3;
140        /// If this control is 1, the logical processor treats specially RDMSR and WRMSR to APIC MSRs (in the range 800H–8FFH). See Section 29.5.
141        const VIRTUALIZED_X2APIC_MODE = 1 << 4;
142        /// If this control is 1, cached translations of linear addresses are associated with a virtualprocessor identifier (VPID). See Section 28.1.
143        const EANBLE_VPID = 1 << 5;
144        /// This control determines whether executions of WBINVD and WBNOINVD cause VM exits.
145        const WBINVD_EXITING = 1 << 6;
146        /// This control determines whether guest software may run in unpaged protected mode or in realaddress mode.
147        const UNRESTRICTED_GUEST = 1 << 7;
148        /// If this control is 1, the logical processor virtualizes certain APIC accesses. See Section 29.4 and Section 29.5.
149        const APIC_REGISTER_VIRTUALIZATION = 1 << 8;
150        /// This controls enables the evaluation and delivery of pending virtual interrupts as well as the
151        /// emulation of writes to the APIC registers that control interrupt prioritization.
152        const VIRTUAL_INTERRUPT_DELIVERY = 1 << 9;
153        /// This control determines whether a series of executions of PAUSE can cause a VM exit
154        /// (see Section 24.6.13 and Section 25.1.3).
155        const PAUSE_LOOP_EXITING = 1 << 10;
156        /// This control determines whether executions of RDRAND cause VM exits.
157        const RDRAND_EXITING = 1 << 1;
158        /// If this control is 0, any execution of INVPCID causes a #UD.
159        const ENABLE_INVPCID = 1 << 12;
160        /// Setting this control to 1 enables use of the VMFUNC instruction in VMX non-root operation. See Section 25.5.6.
161        const ENABLE_VM_FUNCTIONS = 1 << 13;
162        /// If this control is 1, executions of VMREAD and VMWRITE in VMX non-root operation may access
163        /// a shadow VMCS (instead of causing VM exits). See Section 24.10 and Section 30.3.
164        const VMCS_SHADOWING = 1 << 14;
165        /// If this control is 1, executions of ENCLS consult the ENCLS-exiting bitmap to determine whether
166        /// the instruction causes a VM exit. See Section 24.6.16 and Section 25.1.3.
167        const ENABLE_ENCLS_EXITING = 1 << 15;
168        /// This control determines whether executions of RDSEED cause VM exits.
169        const RDSEED_EXITING = 1 << 16;
170        /// If this control is 1, an access to a guest-physical address that sets an EPT dirty bit first adds an
171        /// entry to the page-modification log. See Section 28.3.6.
172        const ENABLE_PML = 1 << 17;
173        /// If this control is 1, EPT violations may cause virtualization exceptions (#VE) instead of VM exits.
174        /// See Section 25.5.7.
175        const EPT_VIOLATION_VE = 1 << 18;
176        /// If this control is 1, Intel Processor Trace suppresses from PIPs an indication that the processor
177        /// was in VMX non-root operation and omits a VMCS packet from any PSB+ produced in VMX nonroot operation (see Chapter 32).
178        const CONCEAL_VMX_FROM_PT = 1 << 19;
179        /// If this control is 0, any execution of XSAVES or XRSTORS causes a #UD.
180        const ENABLE_XSAVES_XRSTORS = 1 << 20;
181        /// If this control is 1, EPT execute permissions are based on whether the linear address being
182        /// accessed is supervisor mode or user mode. See Chapter 28.
183        const MODE_BASED_EXEC_CTL_FOR_EPT = 1 << 22;
184        /// If this control is 1, EPT write permissions may be specified at the granularity of 128 bytes. See
185        /// Section 28.3.4.
186        const SUBPAGE_WRITE_PERM_FOR_EPT = 1 << 23;
187        /// If this control is 1, all output addresses used by Intel Processor Trace are treated as guest
188        /// physical addresses and translated using EPT. See Section 25.5.4.
189        const INTEL_PT_USES_GPA = 1 << 24;
190        /// This control determines whether executions of RDTSC, executions of RDTSCP, and executions
191        /// of RDMSR that read from the IA32_TIME_STAMP_COUNTER MSR return a value modified by the
192        /// TSC multiplier field (see Section 24.6.5 and Section 25.3).
193        const USE_TSC_SCALING = 1 << 25;
194        /// If this control is 0, any execution of TPAUSE, UMONITOR, or UMWAIT causes a #UD.
195        const ENABLE_UWAIT_PAUSE = 1 << 26;
196        /// If this control is 0, any execution of PCONFIG causes a #UD.
197        const ENABLE_PCONFIG = 1 << 27;
198        /// If this control is 1, executions of ENCLV consult the ENCLV-exiting bitmap to
199        const ENABLE_ENCLV_EXITING = 1 << 28;

205    /// Table 24-15. Definitions of VM-Entry Controls.
206    pub struct VmcsEntryCtl: u32 {
207        /// This control determines whether DR7 and the IA32_DEBUGCTL MSR are loaded on VM entry.
208        ///
209        /// The first processors to support the virtual-machine extensions supported only the 1-setting of this control.
210        const LOAD_DEBUG_CTL = 1 << 2;
211        /// On processors that support Intel 64 architecture, this control determines
212        /// whether the logical processor is in IA-32e mode after VM entry.
213        /// Its value is loaded into IA32_EFER.LMA as part of VM entry.1
214        ///
215        /// This control must be 0 on processors that do not support Intel 64 architecture.
216        const IA32E_MODE_GUEST = 1 << 9;
217        /// This control determines whether the logical processor is in system-management mode (SMM) after VM entry.
218        /// This control must be 0 for any VM entry from outside SMM.
219        const ENTRY_TO_SMM = 1 << 10;
220        /// If set to 1, the default treatment of SMIs and SMM is in effect after
221        /// the VM entry (see Section 31.15.7). This control must be 0 for any VM entry from outside SMM.
222        const DEACTIVATE_DUAL_MONITOR_TREATMENT = 1 << 11;
223        /// This control determines whether the IA32_PERF_GLOBAL_CTRL MSR is loaded on VM entry.
224        const LOAD_IA32_PERF_GLOBAL_CTRL = 1 << 13;
225        ///  This control determines whether the IA32_PAT MSR is loaded on VM entry
226        const LOAD_IA32_PAT = 1 << 14;
227        /// This control determines whether the IA32_EFER MSR is loaded on VM entry.
228        const LOAD_IA32_EFER = 1 << 15;
229        /// This control determines whether the IA32_BNDCFGS MSR is loaded on VM entry.
230        const LOAD_IA32_BNDCFGS = 1 << 16;
231        /// If this control is 1, Intel Processor Trace does not produce a paging information packet (PIP)
232        /// on a VM entry or a VMCS packet on a VM entry that returns from SMM (see Chapter 32).
233        const CONCEAL_VMX_FROM_PT = 1 << 17;
234        /// This control determines whether the IA32_RTIT_CTL MSR is loaded on VM entry.
235        const LOAD_IA32_RTIT_CTL = 1 << 18;
236        /// This control determines whether CET-related MSRs and SPP are loaded on VM entry
237        const LOAD_CET_STATE = 1 << 20;
238        /// This control determines whether the IA32_LBR_CTL MSR is loaded on VM entry.
239        const LOAD_GUEST_IA32_LBR_CTL = 1 << 21;
240        /// This control determines whether the IA32_PKRS MSR is loaded on VM entry.
241        const LOAD_PKRS = 1 << 22;
242    }

247    pub struct VmcsExitCtl: u32 {
248        /// This control determines whether DR7 and the IA32_DEBUGCTL MSR are saved on
249        /// VM exit.
250        ///
251        /// The first processors to support the virtual-machine extensions supported only the 1-setting of this control.
252        const SAVE_DEBUG_CTLS = 1 << 2;
253        /// On processors that support Intel 64 architecture, this control determines whether a logical processor
254        /// is in 64-bit mode after the next VM exit. Its value is loaded into CS.L, IA32_EFER.LME, and IA32_EFER.LMA on every VM exit.1
255        /// This control must be 0 on processors that do not support Intel 64 architecture.
256        const HOST_ADDRESS_SPACE_SIZE = 1 << 9;
257        /// This control determines whether the IA32_PERF_GLOBAL_CTRL MSR is loaded on VM exit.
258        const LOAD_IA32_PERF_GLOBAL_CTRL = 1 << 12;
259        /// This control affects VM exits due to external interrupts:
260        ///
261        /// - If such a VM exit occurs and this control is 1, the logical processor acknowledges
262        /// the interrupt controller, acquiring the interrupt’s vector. The vector is stored in
263        /// the VM-exit interruption-information field, which is marked valid.
264        /// - If such a VM exit occurs and this control is 0, the interrupt is not acknowledged and
265        /// the VM-exit interruption-information field is marked invalid.
266        const ACK_INTR_ON_EXIT = 1 << 15;
267        /// This control determines whether the IA32_PAT MSR is saved on VM exit.
268        const SAVE_IA32_PAT = 1 << 18;
269        /// This control determines whether the IA32_PAT MSR is loaded on VM exit.
270        const LOAD_IA32_PAT = 1 << 19;
271        /// This control determines whether the IA32_EFER MSR is saved on VM exit.
272        const SAVE_IA32_EFER = 1 << 20;
273        /// This control determines whether the IA32_EFER MSR is loaded on VM exit.
274        const LOAD_IA32_EFER = 1 << 20;
275        /// This control determines whether the value of the VMX-preemption timer is saved on VM exit.
276        const SAVE_VMX_PREEMPTION_TIMER_VALUE = 1 << 22;
277        /// This control determines whether the IA32_BNDCFGS MSR is cleared on VM exit.
278        const CLEAR_IA32_BNDCFGS = 1 << 23;
279        /// If this control is 1, Intel Processor Trace does not produce a paging information packet (PIP)
280        /// on a VM exit or a VMCS packet on an SMM VM exit (see Chapter 32).
281        const CONCEAL_VMX_FROM_PT = 1 << 24;
282        /// This control determines whether the IA32_RTIT_CTL MSR is cleared on VM exit.
283        const CLEAR_IA32_RTIT_CTL = 1 << 25;
284        /// This control determines whether the IA32_LBR_CTL MSR is cleared on VM exit.
285        const CLEAR_LBR_CTL = 1 << 26;
286        /// This control determines whether CET-related MSRs and SPP are loaded on VM exit.
287        const LOAD_CET_STATE = 1 << 28;
288        /// This control determines whether the IA32_PKRS MSR is loaded on VM exit.
289        const LOAD_PKRS = 1 << 29;
290        /// This control determines whether the IA32_PERF_GLOBAL_CTL MSR is saved on VM exit.
291        const SAVE_IA32_PERF_GLOBAL_CTL = 1 << 30;
292        /// This control determines whether the secondary VM-exit controls are used.
293        /// If this control is 0, the logical processor operates as if all the secondary VM-exit controls were also 0.
294        const ACTIVATE_SECONDARY_CTL = 1 << 31;
