89#[doc(hidden)]
90pub trait Bits {
91    fn bit_test(self, index: usize) -> bool;
92}
93
94impl Bits for u32 {
95    fn bit_test(self, index: usize) -> bool {
96        (self >> index) & 1 != 0
97    }
98}
99
100impl Bits for u64 {
101    fn bit_test(self, index: usize) -> bool {
102        (self >> index) & 1 != 0
103    }
104}

106/// Possible errorkind for Vmx.
107#[derive(Debug)]
108pub enum VmxError {
109    /// Virtual-machine eXtension is not supported.
110    VmxNotSupported,
111    /// Ept is not supported.
112    EptNotSupported,
113    /// Current Cr0 value is invalid.
114    InvalidCr0,
115    /// Current Cr4 value is invalid.
116    InvalidCr4,
117    /// Vmx is disabled in bios.
118    InvalidBiosConfig,
119    /// Vmcs operation has an error.
120    VmxOperationError(vmcs::InstructionError),
121}

123/// Possible errorkind for Vm.
124#[derive(Debug)]
125pub enum VmError {
126    /// Vm operation has error.
127    VmxOperationError(vmcs::InstructionError),
128    /// Failed to handle vmexit.
129    HandleVmexitFailed(ExitReason),
130    /// Controller-private error.
131    ControllerError(Box<dyn core::fmt::Debug + Send + Sync>),
132    /// Failed to decode instruction.
133    FailedToDecodeInstruction,
134    /// Vcpu related error.
135    VCpuError(Box<dyn core::fmt::Debug + Send + Sync>),
136}

138/// Enable the VM-eXtension on this cpu.
139///
140/// # Safety
141/// Must be called once on each cpu.
142pub unsafe fn start_vmx_on_cpu() -> Result<(), VmxError> {
143    unsafe {
144        (Cr4::current() | Cr4::VMXE).apply();
145        // Load vmx realated msrs.
146        let (vmx_cr0_fixed_0, vmx_cr0_fixed_1, vmx_cr4_fixed_0, vmx_cr4_fixed_1) = (
147            Cr0::from_bits_truncate(Msr::<IA32_VMX_CR0_FIXED0>::read()),
148            Cr0::from_bits_truncate(Msr::<IA32_VMX_CR0_FIXED1>::read()),
149            Cr4::from_bits_truncate(Msr::<IA32_VMX_CR4_FIXED0>::read()),
150            Cr4::from_bits_truncate(Msr::<IA32_VMX_CR4_FIXED1>::read()),
151        );
152        let (cr0, cr4) = (Cr0::current(), Cr4::current());
153        // Intel® 64 and IA-32 Architectures Software Developer’s Manual.
154        // 23.8 RESTRICTIONS ON VMX OPERATION
155        if (vmx_cr0_fixed_1 | cr0 != vmx_cr0_fixed_1) || !cr0 & vmx_cr0_fixed_0 != Cr0::empty() {
156            return Err(VmxError::InvalidCr0);
157        }
158        if (vmx_cr4_fixed_1 | cr4 != vmx_cr4_fixed_1) || !cr4 & vmx_cr4_fixed_0 != Cr4::empty() {
159            return Err(VmxError::InvalidCr4);
160        }
161
162        // Intel® 64 and IA-32 Architectures Software Developer’s Manual.
163        // 6.2.1 Detecting and Enabling SMX
164
165        // Try to enable VMX outside SMX operation.
166        let feature_control = Msr::<IA32_FEATURE_CONTROL>::read();
167        if !feature_control.bit_test(2) {
168            Msr::<IA32_FEATURE_CONTROL>::write(feature_control | (1 << 2));
169            if !feature_control.bit_test(2) {
170                return Err(VmxError::InvalidBiosConfig);
171            }
172        }
173
174        // Try to lock.
175        // Lock bit (0 = unlocked, 1 = locked). When set to '1' further writes to this
176        // MSR are blocked
177        let feature_control = Msr::<IA32_FEATURE_CONTROL>::read();
178        if !feature_control.bit_test(0) {
179            Msr::<IA32_FEATURE_CONTROL>::write(feature_control | (1 << 0));
180        }
181
182        // Intel® 64 and IA-32 Architectures Software Developer’s Manual.
183        // 23.6 DISCOVERING SUPPORT FOR VMX
184        if !core::arch::x86_64::__cpuid(1).ecx.bit_test(5) {
185            return Err(VmxError::VmxNotSupported);
186        } else if !Msr::<IA32_VMX_PROC_BASED_CTLS>::read().bit_test(63)
187            || !Msr::<IA32_VMX_PROC_BASED_CTLS>::read().bit_test(33)
188        {
189            return Err(VmxError::EptNotSupported);
190        }
191
192        if cpuid() == 0 {
193            register(100, |_| {});
194        }
195
196        core::mem::ManuallyDrop::new(Box::new(Vmcs::new()))
197            .on()
198            .map_err(VmxError::VmxOperationError)
199    }
200}
