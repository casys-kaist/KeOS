24    _gp: &mut GeneralPurposeRegisters,
25    _launched: &mut bool,

27    naked_asm!(
28        "push rbx",

39        "pop rbp",

112/// Per-vcpu private state.
113pub trait VCpuState
114where
115    Self: Sync + Send + 'static,
116{
117    /// Get enabled pin-based control fields.
118    fn pinbase_ctls(&self) -> VmcsPinBasedVmexecCtl;
119    /// Get enabled proc-based control fields.
120    fn procbase_ctls(&self) -> VmcsProcBasedVmexecCtl;
121    /// Get enabled proc-based secondary control fields.
122    fn procbase_ctls2(&self) -> VmcsProcBasedSecondaryVmexecCtl;
123    /// Get enabled exit control fields.
124    fn exit_ctls(&self) -> VmcsExitCtl;
125    /// Get enabled entry control fields.
126    fn entry_ctls(&self) -> VmcsEntryCtl;

129    /// Handle the vmexit on this vcpu.

131        &mut self,
132        genenric_state: &mut GenericVCpuState,
133    ) -> Result<VmexitResult, VmError>;

136/// A visible state for VCpu.
137pub struct GenericVCpuState<'a> {
138    /// The activated vmcs.
139    pub vmcs: ActiveVmcs,
140    /// general purpose register of the vcpu.
141    pub gprs: &'a mut GeneralPurposeRegisters,
142    /// Weak reference of the vm.
143    pub vm: Weak<dyn VmOps>,
144    // smp id of this vcpu.
145    id: usize,
146    // Pending interrupts.
147    pending_interrupts: &'a [AtomicU64; 4],
148}
149
150impl<'a> GenericVCpuState<'a> {
151    /// Get smp id of this vcpu.
152    #[inline]
153    pub fn id(&self) -> usize {
154        self.id
155    }
156
157    /// Inject the interrupt `vec` into the `active_vmcs`.
158    pub fn inject_interrupt(&self, vec: u8) {
159        // Inject interrupt to the interrupt window
160        let (index, ofs) = (vec / 64, vec & 63);
161        self.pending_interrupts[index as usize].store(1 << ofs, Ordering::SeqCst);
162    }
163}

177    state: S::VcpuState,

197                AtomicU64::new(0),
198            ],
199        }
200    }

203        let Self {
204            vmcs,
205            gprs,
206            vcpu_id,
207            state,
208            launched,
209            vm,
210            pending_interrupts,
211        } = self;
212        Ok(Activated {
213            generic_state: GenericVCpuState {
214                vmcs: Vmcs::activate(vmcs)?,
215                gprs,
216                id: *vcpu_id,
217                vm: vm.clone(),
218                pending_interrupts,
219            },
220            vcpu_state: state,
221            launched,
222            vmcs,
223        })
224    }

236impl<S: VmState + 'static> VCpuOps for SpinLock<VCpu<S>> {
237    fn inject_interrupt(&self, vec: u8) {
238        let (index, ofs) = (vec / 64, vec & 63);
239        let guard = self.lock();
240        guard.pending_interrupts[index as usize].store(1 << ofs, Ordering::SeqCst);
241        guard.unlock();
242    }

252impl<'a, S: VmState + 'static> Activated<'a, S> {
253    pub(crate) unsafe fn init_vcpu(&mut self, exception_bitmap: u32) -> Result<(), VmError> {
254        unsafe {
255            let Self {
256                generic_state: GenericVCpuState { vmcs, .. },
257                vcpu_state,
258                ..
259            } = self;
260            // 26.2.1.1 VM-Execution Control Fields
261            {
262                // Reserved bits in the pin-based VM-execution controls must be set properly.
263                // Software may consult the VMX capability MSRs to determine the
264                // proper settings (see Appendix A.3.1).
265                {
266                    let pinbase_ctls = Msr::<IA32_VMX_PINBASED_CTLS>::read();
267                    let (supported, mut enabled) = (
268                        VmcsPinBasedVmexecCtl::from_bits_unchecked((pinbase_ctls >> 32) as u32),
269                        VmcsPinBasedVmexecCtl::from_bits_unchecked(
270                            pinbase_ctls as u32 & !(VmcsPinBasedVmexecCtl::all().bits()),
271                        ),
272                    );
273                    // enable the guest external interrupt exit
274                    enabled |= vcpu_state.pinbase_ctls();
275                    vmcs.write(
276                        Field::PinBasedExecControls,
277                        (enabled & supported).bits() as u64,
278                    )?;
279                }
280                // Reserved bits in the primary processor-based VM-execution controls must be
281                // set properly. Software may consult the VMX capability MSRs to
282                // determine the proper settings (see Appendix A.3.2).
283                {
284                    let procbase_ctls = Msr::<IA32_VMX_PROC_BASED_CTLS>::read();
285                    let (supported, mut enabled) = (
286                        VmcsProcBasedVmexecCtl::from_bits_unchecked((procbase_ctls >> 32) as u32),
287                        VmcsProcBasedVmexecCtl::from_bits_unchecked(
288                            procbase_ctls as u32 & !(VmcsProcBasedVmexecCtl::all().bits()),
289                        ),
290                    );
291                    // Make sure there are secondary controls.
292                    assert!(supported.contains(VmcsProcBasedVmexecCtl::ACTIVATE_SECONDARY_CTL));
293                    enabled |= VmcsProcBasedVmexecCtl::ACTIVATE_SECONDARY_CTL;
294                    enabled |= vcpu_state.procbase_ctls();
295                    vmcs.write(
296                        Field::ProcessorBasedVmexecControls,
297                        (enabled & supported).bits() as u64,
298                    )?;
299                }
300                // If the “activate secondary controls” primary processor-based VM-execution
301                // control is 1, reserved bits in the secondary processor-based
302                // VM-execution controls must be cleared. Software may consult the VMX
303                // capability MSRs to determine which bits are reserved (see
304                // Appendix A.3.3).
305                {
306                    let procbase_ctls2 = Msr::<IA32_VMX_PROC_BASED_CTLS2>::read();
307                    let (supported, mut enabled) = (
308                        VmcsProcBasedSecondaryVmexecCtl::from_bits_unchecked(
309                            (procbase_ctls2 >> 32) as u32,
310                        ),
311                        VmcsProcBasedSecondaryVmexecCtl::from_bits_unchecked(
312                            procbase_ctls2 as u32
313                                & !(VmcsProcBasedSecondaryVmexecCtl::all().bits()),
314                        ),
315                    );
316                    enabled |= vcpu_state.procbase_ctls2();
317                    vmcs.write(
318                        Field::SecondaryVmexecControls,
319                        (enabled & supported).bits() as u64,
320                    )?;
321                }
322                // 26.2.1.2 VM-Exit Control Fields
323                {
324                    // Reserved bits in the primary VM-exit controls must be set properly.
325                    // Software may consult the VMX capability MSRs to determine the proper settings
326                    // (see Appendix A.4.1).
327                    let exit_ctls = Msr::<IA32_VMX_EXIT_CTLS>::read();
328                    let (supported, mut enabled) = (
329                        VmcsExitCtl::from_bits_unchecked((exit_ctls >> 32) as u32),
330                        VmcsExitCtl::from_bits_unchecked(exit_ctls as u32),
331                    );
332                    enabled |= vcpu_state.exit_ctls();
333                    vmcs.write(Field::VmexitControls, (enabled & supported).bits() as u64)?;
334                }
335                // 26.2.1.3 VM-Entry Control Fields
336                {
337                    // Reserved bits in the VM-entry controls must be set properly.
338                    // Software may consult the VMX capability MSRs to determine the proper settings
339                    // (see Appendix A.5).
340                    let entry_ctls = Msr::<IA32_VMX_ENTRY_CTLS>::read();
341                    let (supported, mut enabled) = (
342                        VmcsEntryCtl::from_bits_unchecked((entry_ctls >> 32) as u32),
343                        VmcsEntryCtl::from_bits_unchecked(entry_ctls as u32),
344                    );
345                    enabled |= vcpu_state.entry_ctls();
346                    vmcs.write(Field::VmentryControls, (supported & enabled).bits() as u64)?;
347                }
348                vmcs.write(Field::ExceptionBitmap, exception_bitmap as u64)?;
349            }
350            // 26.2.2 Checks on Host Control Registers, MSRs, and SSP
351            // 26.2.3 Checks on Host Segment and Descriptor-Table Registers
352            // 26.2.4 Checks Related to Address-Space Size
353            {
354                // The CR0 field must not set any bit to a value not supported in VMX operation
355                // (see Section 23.8).
356                vmcs.write(Field::HostCr0, Cr0::current().bits())?;
357                // The CR4 field must not set any bit to a value not supported in VMX operation
358                // (see Section 23.8).
359                vmcs.write(Field::HostCr4, Cr4::current().bits())?;
360                vmcs.write(Field::HostCr3, read_cr3() as u64)?;
361
362                // Load segments
363                vmcs.write(
364                    Field::HostEsSelector,
365                    Segment::KernelData.into_selector().pack() as u64,
366                )?;
367                vmcs.write(
368                    Field::HostSsSelector,
369                    Segment::KernelData.into_selector().pack() as u64,
370                )?;
371                vmcs.write(
372                    Field::HostDsSelector,
373                    Segment::KernelData.into_selector().pack() as u64,
374                )?;
375                vmcs.write(
376                    Field::HostFsSelector,
377                    Segment::KernelData.into_selector().pack() as u64,
378                )?;
379                vmcs.write(
380                    Field::HostGsSelector,
381                    Segment::KernelData.into_selector().pack() as u64,
382                )?;
383                vmcs.write(
384                    Field::HostCsSelector,
385                    Segment::KernelCode.into_selector().pack() as u64,
386                )?;
387                vmcs.write(
388                    Field::HostTrSelector,
389                    Segment::Tss.into_selector().pack() as u64,
390                )?;
391
392                // Load gdt, Idt
393                vmcs.write(
394                    Field::HostGdtrBase,
395                    SystemTableRegister::new({ &raw const SEGMENT_TABLE }.as_ref().unwrap())
396                        .address,
397                )?;
398                vmcs.write(
399                    Field::HostIdtrBase,
400                    SystemTableRegister::new({ &raw const IDT }.as_ref().unwrap()).address,
401                )?;
402
403                // Load gs, fs, tr
404                vmcs.write(Field::HostFsBase, 0)?;
405                vmcs.write(
406                    Field::HostGsBase,
407                    &mut kernel_gs::KERNEL_GS_BASES[intrinsics::cpuid()] as *mut kernel_gs::KernelGS
408                        as u64,
409                )?;
410                let tss = SegmentTable::current_tss();
411                vmcs.write(Field::HostTrBase, tss as *mut _ as usize as u64)?;
412
413                // Vmexit location
414                vmcs.write(Field::HostRip, vmexit as *const () as usize as u64)?;
415            }
416            vcpu_state.init_guest_state(vmcs)
417        }
418    }

420    pub fn vcpu_loop(&mut self, have_kicked: &AtomicBool) -> Result<VmexitResult, VmError> {
421        assert_eq!(
422            abyss::interrupt::InterruptState::current(),
423            abyss::interrupt::InterruptState::Off
424        );
425        let Self {
426            generic_state,
427            vcpu_state,
428            launched,
429            ..
430        } = self;
431        unsafe {
432            loop {
433                // CHAPTER 26. VM ENTRIES
434                //
435                // Each VM entry performs the following steps in the order indicated:
436                // 1. Basic checks are performed to ensure that VM entry can commence (Section
437                //    26.1).
438                // 2. The control and host-state areas of the VMCS are checked to ensure that
439                //    they are proper for supporting VMX
440                // non-root operation and that the VMCS is correctly configured to support the
441                // next VM exit (Section 26.2).
442                // 3. The following may be performed in parallel or in any order (Section 26.3):
443                // - The guest-state area of the VMCS is checked to ensure that, after the VM
444                //   entry completes, the state of the
445                // logical processor is consistent with IA-32 and Intel 64 architectures.
446                // - Processor state is loaded from the guest-state area and based on controls
447                //   in the VMCS.
448                // - Address-range monitoring is cleared.
449                // 4. MSRs are loaded from the VM-entry MSR-load area (Section 26.4).
450                // 5. If VMLAUNCH is being executed, the launch state of the VMCS is set to
451                //    “launched.”
452                // 6. If the “Intel PT uses guest physical addresses” VM-execution control is 1,
453                //    trace-address pre-translation (TAPT)
454                // may occur (see Section 25.5.4 and Section 26.5).
455                // 7. An event may be injected in the guest context (Section 26.6).
456                //
457                // Steps 1–4 above perform checks that may cause VM entry to fail. Such failures
458                // occur in one of the following three ways:
459                // - Some of the checks in Section 26.1 may generate ordinary faults (for
460                //   example, an invalid-opcode exception).
461                // Such faults are delivered normally.
462                // - Some of the checks in Section 26.1 and all the checks in Section 26.2 cause
463                //   control to pass to the instruction
464                // following the VM-entry instruction. The failure is indicated by setting
465                // RFLAGS.ZF1 (if there is a current VMCS) or RFLAGS.CF (if
466                // there is no current VMCS). If there is a current VMCS, an error number
467                // indicating the cause of the failure is stored in the
468                // VM-instruction error field. See Chapter 30 for the error numbers.
469
470                // Inject pending interrupt if exists.
471                for (index, intr_bitmap) in generic_state.pending_interrupts.iter().enumerate() {
472                    let v = intr_bitmap.load(Ordering::SeqCst);
473                    if v != 0 {
474                        let guest_rflags = Rflags::from_bits_truncate(
475                            generic_state
476                                .vmcs
477                                .read(Field::GuestRflags)
478                                .expect("Failed to read guest rflags."),
479                        );
480                        if guest_rflags.contains(Rflags::IF) {
481                            let ofs = v.trailing_zeros() as usize;
482                            intr_bitmap.fetch_and(!(1 << ofs), Ordering::SeqCst);
483                            let vec = (index * 64 + ofs) as u64;
484                            generic_state
485                                .vmcs
486                                .write(Field::VmentryInterruptionInfo, vec | (1 << 31))
487                                .expect("Failed to set VmentryInterruptionInfo.");
488                        } else {
489                            // We required to wait until Rflags::IF is set. Trap immediatly when it
490                            // becomes 1.
491                            let proc_based_ctls = VmcsProcBasedVmexecCtl::from_bits_unchecked(
492                                generic_state
493                                    .vmcs
494                                    .read(Field::ProcessorBasedVmexecControls)
495                                    .expect("Failed to read vmcs field")
496                                    as u32,
497                            ) | VmcsProcBasedVmexecCtl::INTRWINEXIT;
498                            generic_state
499                                .vmcs
500                                .write(
501                                    Field::ProcessorBasedVmexecControls,
502                                    proc_based_ctls.bits() as u64,
503                                )
504                                .expect("Failed to update ProcessorBasedVmexecControls.");
505                        }
506                        break;
507                    }
508                }
509
510                // Check whether this vcpu is kicked.
511                if have_kicked.load(Ordering::SeqCst) {
512                    return Ok(VmexitResult::Kicked);
513                }
514
515                generic_state.vmcs.write(
516                    Field::HostGsBase,
517                    &mut kernel_gs::KERNEL_GS_BASES[intrinsics::cpuid()] as *mut kernel_gs::KernelGS
518                        as u64,
519                )?;
520                let mut result: i64;
521                {
522                    let mut vcpu_state: i64 = **launched as i64;
523                    core::arch::asm!(
524                        "call {}",
525                        sym vmlaunch_resume,
526                        in("rdi") generic_state.gprs,
527                        inout("rsi") vcpu_state,
528                        lateout("rax") result,
529                        lateout("rcx") _,
530                        lateout("rdx") _,
531                        lateout("rdi") _,
532                        lateout("r8") _,
533                        lateout("r9") _,
534                        lateout("r10") _,
535                        lateout("r11") _,
536                        lateout("r12") _,
537                        lateout("r13") _,
538                        lateout("r14") _,
539                        lateout("r15") _
540                    );
541                    **launched = vcpu_state != 0;
542                }
543
544                match result {
545                    0 => {
546                        let rip = generic_state.vmcs.read(Field::GuestRip)?;
547                        if let Err(err) = match generic_state.vmcs.exit_reason()?.get_basic_reason()
548                        {
549                            BasicExitReason::ExternalInt(Some(ExternalIntInfo {
550                                host_int,
551                                ..
552                            })) => {
553                                return Ok(VmexitResult::ExtInt(*host_int));
554                            }
555                            BasicExitReason::InterruptWindow => {
556                                let proc_based_ctls = VmcsProcBasedVmexecCtl::from_bits_unchecked(
557                                    generic_state
558                                        .vmcs
559                                        .read(Field::ProcessorBasedVmexecControls)
560                                        .expect("Failed to read vmcs field")
561                                        as u32
562                                        & !VmcsProcBasedVmexecCtl::INTRWINEXIT.bits(),
563                                );
564                                generic_state
565                                    .vmcs
566                                    .write(
567                                        Field::ProcessorBasedVmexecControls,
568                                        proc_based_ctls.bits() as u64,
569                                    )
570                                    .expect("Failed to update ProcessorBasedVmexecControls.");
571                                Ok(())
572                            }
573                            _ => match vcpu_state.handle_vmexit(generic_state) {
574                                Ok(VmexitResult::Ok) => Ok(()),
575                                r => return r,
576                            },
577                        } {
578                            println!("err {:?} rip: {:x}", err, rip);
579                            generic_state.vmcs.dump();
580                            return Err(err);
581                        }
582                    }
583                    1 | 2 => return Err(VmError::VmxOperationError(Vmcs::instruction_error())),
584                    _ => unreachable!(),
585                }
586            }
587        }
588    }
589}

591impl<'a, S: VmState> Drop for Activated<'a, S> {
592    fn drop(&mut self) {
593        *self.launched = false;
594        self.vmcs.clear().unwrap();

598/// Possible result of the Vmexit.

602    Ok,
603    /// VCpu is exited.

607    /// This is for internal-control uses.
