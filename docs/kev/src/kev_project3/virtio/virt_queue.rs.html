13/// Command for the virtqueue.
14#[derive(Copy, Clone, Debug, PartialEq)]
15#[repr(u32)]
16pub enum VirtQueueEntryCmd {
17    /// Read
18    Read = 0,
19    /// Write
20    Write = 1,
21}
22
23/// An entry for the virtqueue.
24#[derive(Copy, Clone, Debug)]
25#[repr(C)]
26pub struct VirtQueueEntry {
27    pub addr: Pa,
28    pub size: usize,
29    pub sector: usize,
30    pub cmd: VirtQueueEntryCmd,
31}
32
33/// A container for holding virtqueue.
34#[repr(C)]
35pub struct VirtQueue<T>
36where
37    T: core::ops::Deref<Target = [VirtQueueEntry]>,
38{
39    entries: T,
40}
41
42impl<T> core::ops::Index<usize> for VirtQueue<T>
43where
44    T: core::ops::Deref<Target = [VirtQueueEntry]>,
45{
46    type Output = VirtQueueEntry;
47
48    fn index(&self, index: usize) -> &Self::Output {
49        assert!(index < self.entries.len());
50        unsafe {
51            ((self.entries.as_ptr() as *const _ as usize
52                + core::mem::size_of::<VirtQueueEntry>() * index)
53                as *const VirtQueueEntry)
54                .as_ref()
55                .unwrap()
56        }
57    }
58}
59impl<T> core::ops::IndexMut<usize> for VirtQueue<T>
60where
61    T: core::ops::Deref<Target = [VirtQueueEntry]> + core::ops::DerefMut,
62{
63    fn index_mut(&mut self, index: usize) -> &mut Self::Output {
64        assert!(index < self.entries.len());
65        unsafe {
66            ((self.entries.as_mut_ptr() as *mut _ as usize
67                + core::mem::size_of::<VirtQueueEntry>() * index)
68                as *mut VirtQueueEntry)
69                .as_mut()
70                .unwrap()
71        }
72    }
73}

75impl VirtQueue<Box<[VirtQueueEntry]>> {
76    /// Create a new virtqueue.
77    pub fn new(size: usize) -> Self {
78        let entries = (0..size)
79            .map(|_| VirtQueueEntry {
80                addr: Pa::ZERO,
81                size: 0,
82                sector: 0,
83                cmd: VirtQueueEntryCmd::Read,
84            })
85            .collect::<Vec<_>>()
86            .into_boxed_slice();
87        VirtQueue { entries }
88    }
89    /// Get a virtual address of the virtqueue.
90    pub fn virt_queue_ptr(&self) -> usize {
91        self.entries.as_ptr() as *const _ as usize
92    }
93}
94impl VirtQueue<&'static [VirtQueueEntry]> {
95    /// Get a virtqueue from Kva.
96    ///
97    /// # Safety
98    /// `Kva` must be valid address that points to the virtqueue.
99    pub unsafe fn new_from_raw_ptr(size: usize, queue_kva: Kva) -> Self {
100        let entries = unsafe {
101            core::slice::from_raw_parts(queue_kva.into_usize() as *mut VirtQueueEntry, size)
102        };
103
104        VirtQueue { entries }
105    }
106}

108impl<T> VirtQueue<T>
109where
110    T: core::ops::Deref<Target = [VirtQueueEntry]>,
111{
112    /// Get a fetcher object of the virtqueue.
113    pub fn fetcher<'a>(&'a mut self, mmio: &'a mut VirtIoMmioHeader) -> VirtQueueFetcher<'a, T> {
114        let head = unsafe { read_volatile(&mmio.queue_head as *const u32) as usize };
115        let tail = unsafe { read_volatile(&mmio.queue_tail as *const u32) as usize };
116        VirtQueueFetcher {
117            inner: self,
118            mmio,
119            head,
120            tail,
121        }
122    }
123}

125/// Fetcher object for the virtqueue.
126pub struct VirtQueueFetcher<'a, T>
127where
128    T: core::ops::Deref<Target = [VirtQueueEntry]>,
129{
130    inner: &'a mut VirtQueue<T>,
131    mmio: &'a mut VirtIoMmioHeader,
132    head: usize,
133    tail: usize,
134}
135
136impl<'a, T> VirtQueueFetcher<'a, T>
137where
138    T: core::ops::Deref<Target = [VirtQueueEntry]>,
139{
140    fn charge(&self) -> usize {
141        self.head - self.tail
142    }
143
144    fn size(&self) -> usize {
145        self.inner.entries.len()
146    }
147
148    fn is_empty(&self) -> bool {
149        self.head == self.tail
150    }
151
152    fn is_full(&self) -> bool {
153        self.size() == self.charge()
154    }
155}

157impl<'a> VirtQueueFetcher<'a, &'static [VirtQueueEntry]> {
158    /// Pop a single entry to the virtqueue.
159    pub fn pop_back(&mut self) -> Option<VirtQueueEntry> {
160        if !self.is_empty() {
161            let size = self.size();
162            let r = self.inner.entries[self.tail];
163            self.tail = (self.tail + 1) % size;
164            Some(r)
165        } else {
166            None
167        }
168    }
169    /// Acknowledge the consumed request.
170    pub fn ack(self) -> Result<(), KernelError> {
171        // The sequence of the update in this function
172        // is really important. Do not change the order.
173        unsafe {
174            if read_volatile(&self.mmio.queue_tail) != self.tail as u32 {
175                write_volatile(&mut self.mmio.queue_tail, self.tail as u32);
176            }
177            // This check is required to verify the change we made into mmio area.
178            if read_volatile(&self.mmio.queue_head) == self.head as u32
179                && read_volatile(&self.mmio.queue_tail) == self.tail as u32
180            {
181                if read_volatile(&self.mmio.status) != super::VirtIoStatus::READY as u32 {
182                    return Err(KernelError::IOError);
183                }
184                Ok(())
185            } else {
186                Err(KernelError::IOError)
187            }
188        }
189    }
190}

192impl<'a> VirtQueueFetcher<'a, Box<[VirtQueueEntry]>> {
193    /// Push a single entry to the virtqueue.
194    ///
195    /// This does not ring the doorbell.
196    pub fn push_front(&mut self, value: VirtQueueEntry) -> Result<(), VirtQueueEntry> {
197        if !self.is_full() {
198            let size = self.size();
199            self.inner.entries[self.head] = value;
200            self.head = (self.head + 1) % size;
201            Ok(())
202        } else {
203            Err(value)
204        }
205    }
206
207    /// Kick the doorbell to request commands to the VMM.
208    pub fn kick(mut self) -> Result<(), KernelError> {
209        // The sequence of the update in this function
210        // is really important. Do not change the order.
211        unsafe {
212            if read_volatile(&self.mmio.queue_head) != self.head as u32 {
213                write_volatile(&mut self.mmio.queue_head, self.head as u32);
214                self.tail = read_volatile(&self.mmio.queue_tail) as usize;
215            }
216            // This check is required to verify the change we made into mmio area.
217            if read_volatile(&self.mmio.queue_head) == self.head as u32
218                && read_volatile(&self.mmio.queue_tail) == self.tail as u32
219            {
220                if read_volatile(&self.mmio.status) != super::VirtIoStatus::READY as u32 {
221                    return Err(KernelError::IOError);
222                }
223                Ok(())
224            } else {
225                Err(KernelError::IOError)
226            }
227        }
228    }
229}
