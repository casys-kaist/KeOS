1//! Port-mapped IO vmexit controller.
2//!
3//! Port-mapped I/O is a method of communication between external devices and
4//! the CPU, where each device is assigned a unique port address that is used as
5//! an operand in the in/out instruction family. This allows the CPU to directly
6//! read or write data to the devices. There are different variations of the
7//! in/out instructions, depending on the type and size of the operands used.
8//!
9//! In virtual machine management, the VMM can choose to allow or trap
10//! port-mapped I/O instructions executed by the guest. However, in this
11//! section, all I/O instructions are trapped to the host, giving the VMM
12//! control over their behavior to hand-on the concept of the instruction
13//! emulation.
14//!
15//! Instruction emulation contains two steps; 1) decode the "semantic" of the
16//! instruction; 2) emulate the "semantic" of the instruction. Through the
17//! implementation, you could follow the way that traditional virtualization
18//! emulates the instruction.
19//!
20//! ## Tasks
21//! In this part, You need to emulate the 18 in/out instructions.
22//! The given template conducts the step 2, emulation.
23//!
24//! Decode the instruction and initialize the [`IoInstruction`] instance with
25//! the corresponding opcode and registers. The controller will then use this
26//! information to forward the request to the appropriate handler. When handling
27//! Outsw_DX_m8, Outsw_DX_m16 and Outsd_DX_m32, it is necessary to copy the
28//! memory contents by translating guest virtual address to host virtual
29//! address. You can translate the guest address to the host address by
30//! [`Probe`].

44/// Trait that represent handlers for port-mapped devices.
45pub trait PioHandler
46where
47    Self: Send + Sync,
48{
49    /// handle I/O instructions on the device indicated by the port with the
50    /// operands included in direction.
51    fn handle(
52        &self,
53        port: u16,
54        direction: Direction,
55        p: &dyn Probe,
56        generic_vcpu_state: &mut GenericVCpuState,
57    ) -> Result<VmexitResult, VmError>;
58}
59
60/// Pio vmexit controller.
61pub struct Controller {
62    pios: BTreeMap<u16, Box<dyn PioHandler>>,
63}
64
65impl Controller {
66    /// Create a new pio controller.
67    pub fn new() -> Self {
68        Self {
69            pios: BTreeMap::new(),
70        }
71    }
72    /// Insert pio handler to the index.
73    ///
74    /// Return false if pio handler for index is exists.
75    /// Otherwise, return true.
76    pub fn register(&mut self, port: u16, pio: impl PioHandler + 'static) -> bool {
77        match self.pios.entry(port) {
78            Entry::Occupied(_) => false,
79            Entry::Vacant(v) => {
80                v.insert(Box::new(pio));
81                true
82            }
83        }
84    }
85}
86
87impl kev::vmexits::VmexitController for Controller {
88    fn handle<P: kev::Probe>(
89        &mut self,
90        reason: ExitReason,
91        p: &mut P,
92        generic_vcpu_state: &mut GenericVCpuState,
93    ) -> Result<VmexitResult, kev::VmError> {
94        match reason.get_basic_reason() {
95            BasicExitReason::IoInstruction => {
96                let insn = generic_vcpu_state.vmcs.get_instruction(p)?;
97                self.handle_ioinsn(insn, p, generic_vcpu_state)
98                    .and_then(|s| generic_vcpu_state.vmcs.forward_rip().map(|_| s))
99            }
100            _ => Err(kev::VmError::HandleVmexitFailed(reason)),
101        }
102    }
103}

105#[derive(Debug)]
106/// Possible prefix of io instruction.
107pub enum Prefix {
108    /// Has none of prefix.
109    None,
110    /// Has repe or rep prefix.
111    Rep,
112    /// Has repne.
113    Repne,
114}
115
116#[derive(Debug)]
117/// Direction and the Value of the instruction
118pub enum Direction {
119    /// Input byte from I/O port into AL.
120    InbAl,
121    /// Input word from I/O port into AX.
122    InwAx,
123    /// Input double word from I/O port into EAX.
124    IndEax,
125    /// Input byte from I/O port into memory.
126    Inbm(Gva),
127    /// Input word from I/O port into memory.
128    Inwm(Gva),
129    /// Input double word from I/O port into memory.
130    Indm(Gva),
131    /// Output a byte (1 byte)
132    Outb(u8),
133    /// Output a word (2 bytes)
134    Outw(u16),
135    /// Output a double word (4 bytes)
136    Outd(u32),
137}

139#[derive(Debug)]
140/// The decoded information of io instruction.
141pub struct IoInstruction {
142    /// Port of the io iostruction
143    pub port: u16,
144    /// Direction and value of the io instruction
145    pub direction: Direction,
146}

148impl Controller {
149    fn handle_ioinsn_one<P: Probe>(
150        &self,
151        insn: Instruction,
152        p: &mut P,
153        generic_vcpu_state: &mut GenericVCpuState,
154    ) -> Result<VmexitResult, VmError> {
155        // Hint
156        //   - Use [`Probe::gva2hva`] to translate guest memory to host memory.
157        let IoInstruction { port, direction } = match insn.code() {
158            // -- in families.
159            // in al, dx
160            // Input byte from I/O port in DX into AL.
161            Code::In_AL_DX => todo!(),
162            // in ax, dx
163            // Input word from I/O port in DX into AX.
164            Code::In_AX_DX => todo!(),
165            // in eax, dx
166            // Input doubleword from I/O port in DX into EAX.
167            Code::In_EAX_DX => todo!(),
168            // in al, imm8
169            // Input byte from imm8 I/O port address into AL.
170            Code::In_AL_imm8 => todo!(),
171            // in ax, imm8
172            // Input word from imm8 I/O port address into AX.
173            Code::In_AX_imm8 => todo!(),
174            // in eax, imm8
175            // Input dword from imm8 I/O port address into EAX.
176            Code::In_EAX_imm8 => todo!(),
177            // insb
178            // Input byte from I/O port specified in DX into memory location specified in ES:(E)DI
179            // or RDI.*
180            Code::Insb_m8_DX => todo!(),
181            // insw
182            // Input word from I/O port specified in DX into memory location specified in ES:(E)DI
183            // or RDI.1
184            Code::Insw_m16_DX => todo!(),
185            // insd
186            // Input doubleword from I/O port specified in DX into memory location specified in
187            // ES:(E)DI or RDI.1
188            Code::Insd_m32_DX => todo!(),
189            // -- out families.
190            // out dx, al
191            // Output byte from AL into I/O port in DX.
192            Code::Out_DX_AL => todo!(),
193            // out dx, ax
194            // Output word from AX into I/O port in DX.
195            Code::Out_DX_AX => todo!(),
196            // out dx, eax
197            // Output double word from AX into I/O port in DX.
198            Code::Out_DX_EAX => todo!(),
199            // out imm8, al
200            // Output byte from AL into I/O port in imm8.
201            Code::Out_imm8_AL => todo!(),
202            // out imm8, ax
203            // Output byte from AL into I/O port in imm8.
204            Code::Out_imm8_AX => todo!(),
205            // out imm8, eax
206            // Output double word from EAX into I/O port in imm8.
207            Code::Out_imm8_EAX => todo!(),
208            // outsb
209            // Output byte from memory location specified in DS:(E)SI or RSI to I/O port specified
210            // in DX**.
211            Code::Outsb_DX_m8 => todo!(),
212            // outsw
213            // Output word from memory location specified in DS:(E)SI or RSI to I/O port specified
214            // in DX**.
215            Code::Outsw_DX_m16 => todo!(),
216            // outsd
217            // Output doubleword from memory location specified in DS:(E)SI or RSI to I/O port
218            // specified in DX**.
219            Code::Outsd_DX_m32 => todo!(),
220            _ => unreachable!(),
221        };
222        let result = if let Some(handler) = self.pios.get(&port) {
223            handler.handle(port, direction, p, generic_vcpu_state)
224        } else {
225            Err(VmError::ControllerError(Box::new(format!(
226                "Unknown io port: 0x{port:x}"
227            ))))
228        };
229        let df = Rflags::from_bits_truncate(generic_vcpu_state.vmcs.read(Field::GuestRflags)?)
230            .contains(Rflags::DF);
231        match insn.code() {
232            Code::Insb_m8_DX if !df => {
233                generic_vcpu_state.gprs.rdi = generic_vcpu_state.gprs.rdi.overflowing_add(1).0
234            }
235            Code::Insb_m8_DX if df => {
236                generic_vcpu_state.gprs.rdi = generic_vcpu_state.gprs.rdi.overflowing_sub(1).0
237            }
238            Code::Insw_m16_DX if !df => {
239                generic_vcpu_state.gprs.rdi = generic_vcpu_state.gprs.rdi.overflowing_add(2).0
240            }
241            Code::Insw_m16_DX if df => {
242                generic_vcpu_state.gprs.rdi = generic_vcpu_state.gprs.rdi.overflowing_sub(2).0
243            }
244            Code::Insd_m32_DX if !df => {
245                generic_vcpu_state.gprs.rdi = generic_vcpu_state.gprs.rdi.overflowing_add(4).0
246            }
247            Code::Insd_m32_DX if df => {
248                generic_vcpu_state.gprs.rdi = generic_vcpu_state.gprs.rdi.overflowing_sub(4).0
249            }
250            Code::Outsb_DX_m8 if !df => {
251                generic_vcpu_state.gprs.rsi = generic_vcpu_state.gprs.rsi.overflowing_add(1).0
252            }
253            Code::Outsb_DX_m8 if df => {
254                generic_vcpu_state.gprs.rsi = generic_vcpu_state.gprs.rsi.overflowing_sub(1).0
255            }
256            Code::Outsw_DX_m16 if !df => {
257                generic_vcpu_state.gprs.rsi = generic_vcpu_state.gprs.rsi.overflowing_add(2).0
258            }
259            Code::Outsw_DX_m16 if df => {
260                generic_vcpu_state.gprs.rsi = generic_vcpu_state.gprs.rsi.overflowing_sub(2).0
261            }
262            Code::Outsd_DX_m32 if !df => {
263                generic_vcpu_state.gprs.rsi = generic_vcpu_state.gprs.rsi.overflowing_add(4).0
264            }
265            Code::Outsd_DX_m32 if df => {
266                generic_vcpu_state.gprs.rsi = generic_vcpu_state.gprs.rsi.overflowing_sub(4).0
267            }
268            _ => (),
269        }
270        result
271    }
272
273    fn handle_ioinsn<P: Probe>(
274        &self,
275        insn: Instruction,
276        p: &mut P,
277        generic_vcpu_state: &mut GenericVCpuState,
278    ) -> Result<VmexitResult, VmError> {
279        if insn.has_rep_prefix() || insn.has_repne_prefix() {
280            while generic_vcpu_state.gprs.rcx != 0 {
281                let result = self.handle_ioinsn_one(insn, p, generic_vcpu_state);
282                generic_vcpu_state.gprs.rcx -= 1;
283                match result {
284                    Ok(VmexitResult::Ok) => (),
285                    r => return r,
286                }
287            }
288            Ok(VmexitResult::Ok)
289        } else {
290            self.handle_ioinsn_one(insn, p, generic_vcpu_state)
291        }
292    }
