1//! Virtual machine control structure.
2//!
3//! Virtual machine control data structures (VMCS) are structures used by x86
4//! during virtual machine execution. VMCS stores detailed information on how a
5//! virtual machine will operate, including its CPU and memory, IO
6//! configurations, interrupt handling, and other hardware settings.
7//! The configuration for the VMCS in Project 2 is called "no_ept_vm." This
8//! setup does not virtualize the memory, and instead directly maps guest
9//! virtual addresses to the corresponding host virtual addresses.
10//!
11//! ## Tasks
12//! In this project, you are required to configure the guest CR3 and RIP
13//! registers. The guest page table is stored at the
14//! vbsp_vcpu_state.mem.page_table. Setting the guest RIP register should be set
15//! to the entry point of the guest virtual machine.
16//!
17//! If you implement the page table in project 1 incorrectly, even if you pass
18//! the test cases, then the virtual machine will not operate properly.

44/// The Vmstate of NoEptVmState.
45#[derive(Default)]
46pub struct NoEptVmState {
47    code: &'static [u8],
48}

50/// Error for setup_vbsp.
51#[derive(Debug)]
52pub enum Error {
53    /// Error that occurs when page table mapping failed.
54    PageTableMappingError(PageTableMappingError),
55    /// Error that occurs when vmwrite instruction failed.
56    VmError(VmError),
57}

59impl NoEptVmState {
60    /// Create a new instance of NoEptVmState
61    pub fn new(code: &'static [u8]) -> Self {
62        Self { code }
63    }

66impl kev::vm::VmState for NoEptVmState {
67    type VcpuState = NoEptVcpuState;
68    type Error = Error;

70    fn vcpu_state(&self) -> Self::VcpuState {
71        let (mut pio_ctl, hypercall_ctl, cpuid_ctl, mut msr_ctl) = (
72            pio::Controller::new(),
73            hypercall::Controller::new(HypercallCtx),
74            cpuid::Controller::new(),
75            msr::Controller::new(),
76        );
77        pio_ctl.register(3, crate::pio::PioHandlerDummy);
78        pio_ctl.register(0x3f8, crate::pio::PioHandlerPrint);
79        pio_ctl.register(0xbb, crate::pio::PioHandlerQueue::new());
80
81        assert!(msr_ctl.insert(0xabc, crate::msr::StackMsr::new()));
82
83        NoEptVcpuState {
84            mem: NoEpt {
85                page_table: PageTable(PageTableRoot::new_boxed()),
86            },
87            vmexit_controller: (pio_ctl, (hypercall_ctl, (cpuid_ctl, msr_ctl))),
88        }
89    }

91    fn setup_vbsp(
92        &self,
93        vbsp_generic_state: &mut GenericVCpuState,
94        vbsp_vcpu_state: &mut Self::VcpuState,
95    ) -> Result<(), Self::Error> {
96        const ENTRY: Va = Va::new(0x4000).unwrap();
97        const WRITABLE: Va = Va::new(0x2000).unwrap();
98
99        // allocate a page to be written by guest
100        vbsp_vcpu_state
101            .mem
102            .page_table
103            .map(
104                WRITABLE,
105                Page::new(),
106                Permission::READ | Permission::WRITE | Permission::EXECUTABLE,
107            )
108            .map_err(Error::PageTableMappingError)?;
109
110        // Prepare code pages.
111        let mut pgs = keos::mm::ContigPages::new((self.code.len() + 0xfff) & !0xfff)
112            .expect("Failed to allocate code page")
113            .split();
114        // Copy code to page
115        let ((srcs, remainder), mut dsts) = (self.code.as_chunks::<0x1000>(), pgs.iter_mut());
116        for src in srcs {
117            let dst = dsts.next().unwrap();
118            dst.inner_mut().copy_from_slice(src);
119        }
120        let dst = dsts.next().unwrap();
121        dst.inner_mut()[..remainder.len()].copy_from_slice(remainder);
122
123        // Map into page table.
124        let mut base = ENTRY;
125        for pg in pgs.into_iter() {
126            vbsp_vcpu_state
127                .mem
128                .page_table
129                .map(base, pg, Permission::READ | Permission::EXECUTABLE)
130                .map_err(Error::PageTableMappingError)?;
131            base += 0x1000;
132        }
133        let gdt = SystemTableRegister::new(unsafe { (&raw const SEGMENT_TABLE).as_ref().unwrap() });
134        let base = gdt.address & !(PAGE_MASK as u64);
135        for kva in (base..gdt.address + gdt.size as u64).step_by(0x1000) {
136            let kva = Kva::new(kva as usize).unwrap();
137            let mut page = Page::new();
138            page.inner_mut().copy_from_slice(unsafe {
139                core::slice::from_raw_parts(kva.into_usize() as *const u8, 0x1000)
140            });
141            vbsp_vcpu_state
142                .mem
143                .page_table
144                .map(kva.into_va(), page, Permission::READ)
145                .map_err(Error::PageTableMappingError)?;
146        }
147        // Run a guest on 64bit mode directly.
148        let vmcs = &vbsp_generic_state.vmcs;
149        vmcs.write(
150            Field::GuestCsSelector,
151            Segment::KernelCode.into_selector().pack() as u64,
152        )
153        .map_err(Error::VmError)?;
154        vmcs.write(
155            Field::GuestEsSelector,
156            Segment::KernelData.into_selector().pack() as u64,
157        )
158        .map_err(Error::VmError)?;
159        vmcs.write(
160            Field::GuestSsSelector,
161            Segment::KernelData.into_selector().pack() as u64,
162        )
163        .map_err(Error::VmError)?;
164        vmcs.write(
165            Field::GuestDsSelector,
166            Segment::KernelData.into_selector().pack() as u64,
167        )
168        .map_err(Error::VmError)?;
169        vmcs.write(
170            Field::GuestFsSelector,
171            Segment::KernelData.into_selector().pack() as u64,
172        )
173        .map_err(Error::VmError)?;
174        vmcs.write(
175            Field::GuestGsSelector,
176            Segment::KernelData.into_selector().pack() as u64,
177        )
178        .map_err(Error::VmError)?;
179        vmcs.write(
180            Field::GuestTrSelector,
181            Segment::KernelData.into_selector().pack() as u64,
182        )
183        .map_err(Error::VmError)?;
184        vmcs.write(Field::GuestLdtrSelector, 0)
185            .map_err(Error::VmError)?;
186
187        vmcs.write(Field::GuestCsBase, 0).map_err(Error::VmError)?;
188        vmcs.write(Field::GuestEsBase, 0).map_err(Error::VmError)?;
189        vmcs.write(Field::GuestSsBase, 0).map_err(Error::VmError)?;
190        vmcs.write(Field::GuestDsBase, 0).map_err(Error::VmError)?;
191        vmcs.write(Field::GuestFsBase, 0).map_err(Error::VmError)?;
192        vmcs.write(Field::GuestGsBase, 0).map_err(Error::VmError)?;
193        vmcs.write(Field::GuestLdtrBase, 0)
194            .map_err(Error::VmError)?;
195        vmcs.write(
196            Field::GuestGdtrBase,
197            SystemTableRegister::new(unsafe { (&raw const SEGMENT_TABLE).as_ref().unwrap() })
198                .address,
199        )
200        .map_err(Error::VmError)?;
201        vmcs.write(Field::GuestIdtrBase, 0)
202            .map_err(Error::VmError)?;
203        vmcs.write(Field::GuestTrBase, 0).map_err(Error::VmError)?;
204
205        vmcs.write(Field::GuestCsLimit, u64::MAX)
206            .map_err(Error::VmError)?;
207        vmcs.write(Field::GuestEsLimit, u64::MAX)
208            .map_err(Error::VmError)?;
209        vmcs.write(Field::GuestSsLimit, u64::MAX)
210            .map_err(Error::VmError)?;
211        vmcs.write(Field::GuestDsLimit, u64::MAX)
212            .map_err(Error::VmError)?;
213        vmcs.write(Field::GuestFsLimit, u64::MAX)
214            .map_err(Error::VmError)?;
215        vmcs.write(Field::GuestGsLimit, u64::MAX)
216            .map_err(Error::VmError)?;
217        vmcs.write(Field::GuestLdtrLimit, 0xffff)
218            .map_err(Error::VmError)?;
219        vmcs.write(Field::GuestGdtrLimit, 0xffff)
220            .map_err(Error::VmError)?;
221        vmcs.write(Field::GuestIdtrLimit, 0xffff)
222            .map_err(Error::VmError)?;
223        vmcs.write(Field::GuestTrLimit, 0x67)
224            .map_err(Error::VmError)?;
225
226        vmcs.write(Field::GuestCsAccessRights, 0xa09b)
227            .map_err(Error::VmError)?;
228        vmcs.write(Field::GuestEsAccessRights, 0xc093)
229            .map_err(Error::VmError)?;
230        vmcs.write(Field::GuestSsAccessRights, 0xc093)
231            .map_err(Error::VmError)?;
232        vmcs.write(Field::GuestDsAccessRights, 0xc093)
233            .map_err(Error::VmError)?;
234        vmcs.write(Field::GuestFsAccessRights, 0xc093)
235            .map_err(Error::VmError)?;
236        vmcs.write(Field::GuestGsAccessRights, 0xc093)
237            .map_err(Error::VmError)?;
238        vmcs.write(Field::GuestTrAccessRights, 0x8b)
239            .map_err(Error::VmError)?;
240        vmcs.write(Field::GuestLdtrAccessRights, 0x10000)
241            .map_err(Error::VmError)?;
242
243        vmcs.write(Field::GuestActivityState, 0)
244            .map_err(Error::VmError)?;
245        vmcs.write(Field::GuestInterruptibilityState, 0)
246            .map_err(Error::VmError)?;
247
248        vmcs.write(Field::GuestCr0, Cr0::current().bits())
249            .map_err(Error::VmError)?;
250        vmcs.write(Field::GuestCr4, Cr4::current().bits())
251            .map_err(Error::VmError)?;
252        vmcs.write(Field::GuestIa32Efer, 0x500)
253            .map_err(Error::VmError)?;
254
255        vmcs.write(Field::GuestLinkPointer, 0xffff_ffff)
256            .map_err(Error::VmError)?;
257        vmcs.write(Field::GuestLinkPointerHi, 0xffff_ffff)
258            .map_err(Error::VmError)?;
259
260        vmcs.write(Field::GuestDr7, 0).map_err(Error::VmError)?;
261        vmcs.write(Field::GuestIa32Debugctl, 0)
262            .map_err(Error::VmError)?;
263
264        vmcs.write(Field::GuestRflags, Rflags::_1.bits())
265            .map_err(Error::VmError)?;
266
267        vmcs.write(
268            Field::GuestCr3,
269            vbsp_vcpu_state.mem.page_table.pa().into_usize() as u64,
270        )
271        .map_err(Error::VmError)?;
272
273        vbsp_generic_state
274            .vmcs
275            .write(Field::GuestRip, ENTRY.into_usize() as u64)
276            .map_err(Error::VmError)?;
277        Ok(())
278    }

281struct NoEpt {
282    page_table: PageTable,
283}
284impl kev::Probe for NoEpt {
285    fn gpa2hpa(&self, _vmcs: &ActiveVmcs, _gpa: Gpa) -> Option<Pa> {
286        // Because the term `gpa` is not existed when ept is not existed, we cannot be
287        // reachable here.
288        unreachable!()
289    }
290    fn gva2hpa(&self, _vmcs: &ActiveVmcs, gva: Gva) -> Option<Pa> {
291        // In this type of VM, gva is same as hva.
292        let (va, ofs) = {
293            let gva = gva.into_usize();
294            (Va::new(gva & !PAGE_MASK)?, gva & PAGE_MASK)
295        };
296        self.page_table
297            .walk(va)
298            .map(|pte| pte.pa())
299            .ok()?
300            .map(|pa| pa + ofs)
301    }
302}

316impl kev::vcpu::VCpuState for NoEptVcpuState {
317    fn pinbase_ctls(&self) -> VmcsPinBasedVmexecCtl {
318        VmcsPinBasedVmexecCtl::EXTERNAL_INTERRUPT_EXITING
319    }
320    fn procbase_ctls(&self) -> VmcsProcBasedVmexecCtl {
321        VmcsProcBasedVmexecCtl::HLT_EXITING
322            | VmcsProcBasedVmexecCtl::CR3LOADEXIT
323            | VmcsProcBasedVmexecCtl::UNCONDIOEXIT
324    }
325    fn procbase_ctls2(&self) -> VmcsProcBasedSecondaryVmexecCtl {
326        VmcsProcBasedSecondaryVmexecCtl::ENABLE_RDTSCP
327    }
328    fn entry_ctls(&self) -> VmcsEntryCtl {
329        VmcsEntryCtl::IA32E_MODE_GUEST | VmcsEntryCtl::LOAD_IA32_EFER
330    }
331    fn exit_ctls(&self) -> VmcsExitCtl {
332        VmcsExitCtl::HOST_ADDRESS_SPACE_SIZE | VmcsExitCtl::ACK_INTR_ON_EXIT
333    }
334    fn init_guest_state(&self, _vmcs: &ActiveVmcs) -> Result<(), VmError> {
335        Ok(())
336    }
337
338    fn handle_vmexit(
339        &mut self,
340        generic_vcpu_state: &mut GenericVCpuState,
341    ) -> Result<VmexitResult, VmError> {
342        // Hint: Use Vmcs.
343        let exit_reason = generic_vcpu_state
344            .vmcs
345            .exit_reason()
346            .expect("unexpected vmexit.");
347        let Self {
348            mem,
349            vmexit_controller,
350        } = self;
351        vmexit_controller.handle(exit_reason, mem, generic_vcpu_state)
352    }
